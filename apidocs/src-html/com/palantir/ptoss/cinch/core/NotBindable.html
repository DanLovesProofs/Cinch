014    package com.palantir.ptoss.cinch.core;
015    
016    import java.lang.annotation.ElementType;
017    import java.lang.annotation.Retention;
018    import java.lang.annotation.RetentionPolicy;
019    import java.lang.annotation.Target;
020    
021    /**
022     * Marks the annotated field as not bindable.  In Cinch, all bindable fields are required to be
023     * marked <code>final</code>.  In fact, if a {@link BindableModel} has a field that is not marked
024     * <code>final</code>, a runtime error will during the call to {@link Bindings#bind(Object)}.  To 
025     * allow {@link BindableModel}s to have non-final fields, mark those fields with this annotation.
026     * <p>  
027     * It can be thought of as similar to {@link SuppressWarnings} as a way to force the coder
028     * to explicitly specify intention when doing something dangerous.
029     * </p>
030     */
031    @Retention(RetentionPolicy.RUNTIME)
032    @Target({ElementType.FIELD})
033    public @interface NotBindable {
034            // Empty annotation.
035    }
