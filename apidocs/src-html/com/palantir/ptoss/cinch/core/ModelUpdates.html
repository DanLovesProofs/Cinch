001    package com.palantir.ptoss.cinch.core;
002    
003    import com.palantir.ptoss.cinch.swing.Bound;
004    
005    /**
006     * Pre-defined {@link ModelUpdate} types with special meanings to the framework.
007     *
008     * @see Bound
009     * @see CallOnUpdate
010     */
011    public enum ModelUpdates implements ModelUpdate {
012            
013            /**
014             * <p>
015             * This is used for unspecified update types in the default implementations of {@link AbstractListenableModel}.
016             * Listeners that don't have a specified "on" parameter should react to these model updates.
017             * </p>
018             * <p>
019             * <strong>Note:</strong> when bound to a specific {@link ModelUpdate}, a control will not
020             * receive updates not sent specifically to any {@link ModelUpdate} - i.e. default updates. To
021             * selectively answer a specific {@link ModelUpdate} and also still get default notications,
022             * add {@link ModelUpdates#UNSPECIFIED} to the list of {@link ModelUpdate}s passed to {@link Bound#on()}.
023             * </p>
024             * @see Bound
025             */
026            UNSPECIFIED,
027            
028            /**
029             * This is used to sync all listeners.  All listeners, regardless of whether they have an "on" parameter, should
030             * react to these model updates. 
031             */
032            ALL;
033    }
