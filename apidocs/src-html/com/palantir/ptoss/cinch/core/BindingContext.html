044    /**
045     * <p>A {@link BindingContext} holds information about how to bind various parts of a Java Object.
046     * 
047     * <p><b>Binding Constants</b> - TODO

055    public class BindingContext {
056            /**
057             * The object for which the context has been built.
058             */
059            private final Object object;
060            
061            /**
062             * All of the fields of type {@link BindableModel} on the object.
063             * @see #indexBindableModels()
064             */
065            private final Map<String, Field> bindableModels;
066            
067            /**
068             * All of the bindable methods on the object.
069             * @see #indexBindableMethods()
070             */
071            private final Map<String, ObjectFieldMethod> bindableMethods;
072            
073            /**
074             * @see #indexBindableModelMethods()
075             */
076            private final Map<String, ObjectFieldMethod> bindableModelMethods;
077            
078            /**
079             * The map from all static final field names on the object to the objects contained in those
080             * fields.
081             * @see #indexBindableConstants()
082             */
083            private final Map<String, Object> bindableConstants;
084            
085            /**
086             * All of the getters available on the bindable models.
087             */
088            private final Map<String, ObjectFieldMethod> bindableGetters;
089            
090            /**
091             * All of the setters available on the bindable models.
092             */
093            private final Map<String, ObjectFieldMethod> bindableSetters;

095            /**
096             * Create a BindingContext for the given, non-null object.  Throws a {@link BindingException}
097             * if there is a problem.
098             * @param object the object - cannot be null
099             */
100            public BindingContext(Object object) {
101                    Validate.notNull(object);
102                    this.object = object;
103                    try {
104                            bindableModels = indexBindableModels();
105                    bindableMethods = indexBindableMethods();
106                    bindableModelMethods = indexBindableModelMethods();
107                    bindableConstants = indexBindableConstants();
108                    bindableGetters = indexBindableProperties(Reflections.getterFunction(PropertyDescriptor.class, Method.class, "readMethod"));
109                    bindableSetters = indexBindableProperties(Reflections.getterFunction(PropertyDescriptor.class, Method.class, "writeMethod"));
110            } catch (Exception e) {
111                    throw new BindingException("could not create BindingContext", e); 
112            }
113            }

115            /**
116             * Gets a constant from the binding context.  Constants are static, final fields of the bound
117             * object.
118             * 
119             * @param key the name of the field
120             * @return the value of the field
121             */
122            public Object getBindableConstant(String key) {
123                    return bindableConstants.get(key);
124            }

126            /**
127             * Look through all of the declared, static, final fields of the context object, grab the value,
128             * and insert a mapping from the field's name to the object.
129             * 
130             * Note that this will index non-public fields.
131             * 
132             * @return the bindable constants map
133             * @throws IllegalArgumentException on reflection error
134             * @throws IllegalAccessException on reflection error
135             */
136            private Map<String, Object> indexBindableConstants() throws IllegalArgumentException, IllegalAccessException {
137                    Map<String, Object> map = Maps.newHashMap();
138                    for (Field field : object.getClass().getDeclaredFields()) {
139                            boolean accessible = field.isAccessible();
140                            field.setAccessible(true);
141                            if (Reflections.isFieldFinal(field) && Reflections.isFieldStatic(field)) {
142                                    map.put(field.getName(), field.get(object));
143                            }
144                            field.setAccessible(accessible);
145                    }
146                return map;
147        }

195            /**
196             * Returns the list of {@link ModelUpdate} types in this binding context. 
197             * @param modelClass
198             * @return the of {@link Class}es that implement {@link ModelUpdate} in this binding context.
199             */
200            public static List<Class<?>> findModelUpdateClass(final BindableModel modelClass) {
201                    List<Class<?>> classes = Reflections.getTypesOfTypeForClassHierarchy(
202                                    modelClass.getClass(), ModelUpdate.class);
203                    Predicate<Class<?>> isEnum = new Predicate<Class<?>>() {
204                            public boolean apply(final Class<?> input) {
205                        return input.isEnum();
206                }
207                    };
208                    // Look for ModelUpdate classes in implemented interfaces
209                    classes = Lists.newArrayList(Iterables.filter(classes, isEnum));
210                    for (Class<?> iface : modelClass.getClass().getInterfaces()) {
211                            classes.addAll(Lists.newArrayList(Iterables.filter(
212                                            Reflections.getTypesOfTypeForClassHierarchy(
213                                            iface, ModelUpdate.class), isEnum)));
214                    }
215                    if (classes.size() == 0) {
216                            return null;
217                    }
218                    return classes;
219            }

221            /**
222             * Resolves a string reference, as specified in the <code>on</code> parameter of
223             * a {@link Bound} annotation to an Enum object in this runtime.
224             * @param on <code>on</code> parameter from a {@link Bound} annotation.
225             * @param model 
226             * @return the resolved object
227             * @throws IllegalArgumentException if the referenced object can't be found.
228             */
229            public static ModelUpdate findOnObject(final String on, final BindableModel model) {
230                    ModelUpdate onObject = null;
231                    if (on != null && on.trim().length() > 0) {
232                            final List<Class<?>> updateClasses = findModelUpdateClass(model);
233                            for (Class<?> updateClass : updateClasses) {
234                                    try {
235                                            onObject = (ModelUpdate)Reflections.evalEnum(updateClass, on);
236                                            return onObject;
237                                    } catch (IllegalArgumentException e) {
238                                            // swallow this if we don't find the enum on one of the
239                                            // classes, continue to next class.
240                                    }
241                            }
242                            throw new IllegalArgumentException("could not find \"on\" parameter " + on);
243                    }
244                    return onObject;
245            }

247            private Map<String, ObjectFieldMethod> indexBindableProperties(Function<PropertyDescriptor, Method> methodFn) throws IntrospectionException {
248                    final Map<ObjectFieldMethod, String> getterOfms = Maps.newHashMap();
249                    for (Field field : Sets.newHashSet(bindableModels.values())) {
250                            BeanInfo beanInfo = Introspector.getBeanInfo(field.getType());
251                        PropertyDescriptor[] props = beanInfo.getPropertyDescriptors();
252                        for (PropertyDescriptor descriptor : props) {
253                            Method method = methodFn.apply(descriptor);
254                            if (method == null) {
255                                    continue;
256                            }
257                            BindableModel model = getFieldObject(field, BindableModel.class);
258                            getterOfms.put(new ObjectFieldMethod(model, field, method), descriptor.getName());
259                        }
260                    }
261                    return dotIndex(getterOfms.keySet(), ObjectFieldMethod.TO_FIELD_NAME, Functions.forMap(getterOfms));
262            }
263            
264            private static <T> Map<String, T> dotIndex(Collection<T> items, Function<T, String> qualifierFn, Function<T, String> blindFn) {
265                    Set<String> ambiguousNames = Sets.newHashSet();
266                    Map<String, T> results = Maps.newHashMap();
267                    for (T item : items) {
268                            String blindKey = blindFn.apply(item);
269                            if (!ambiguousNames.contains(blindKey)) {
270                                    if (results.containsKey(blindKey)) {
271                                            results.remove(blindKey);
272                                            ambiguousNames.add(blindKey);
273                                    } else {
274                                            results.put(blindKey, item);
275                                    }
276                            }
277                            String qualifiedKey = qualifierFn.apply(item) + "." + blindKey;
278                            results.put(qualifiedKey, item);
279                    }
280                    return results;
281            }

291            public Set<BindableModel> getBindableModels() {
292                    Function<Field, BindableModel> f = new Function<Field, BindableModel>() {
293                            public BindableModel apply(Field from) {
294                                    return getFieldObject(from, BindableModel.class);
295                            }
296                    };
297                    return ImmutableSet.copyOf(Iterables.transform(bindableModels.values(), f));
298            }

301                    return Reflections.getAnnotatedFieldsForClassHierarchy(object.getClass(), klass);
302            }

304            public List<ObjectFieldMethod> getAnnotatedParameterlessMethods(final Class<? extends Annotation> annotation) {
305                    return Lists.newArrayList(Iterables.filter(Reflections.getParameterlessMethodsForClassHierarchy(object), 
306                                    new Predicate<ObjectFieldMethod>() {
307                                            public boolean apply(ObjectFieldMethod input) {
308                                                    return input.getMethod().isAnnotationPresent(annotation);
309                        }
310                                    }));
311            }

313            private static List<ObjectFieldMethod> getParameterlessMethods(Object object, Field field) {
314            List<ObjectFieldMethod> methods = Lists.newArrayList();
315            for (Method method : field.getType().getDeclaredMethods()) {
316                    if (method.getParameterTypes().length == 0 && Reflections.isMethodPublic(method)) {
317                            methods.add(new ObjectFieldMethod(object, field, method));
318                    }
319            }
320            return methods;
321        }
322            
323            private static List<ObjectFieldMethod> getParameterlessMethodsOnFieldTypes(Object object, List<Field> fields) throws IllegalArgumentException {
324                    List<ObjectFieldMethod> methods = Lists.newArrayList();
325                    for (Field field : fields) {
326                            Object fieldObject = Reflections.getFieldObject(object, field, Object.class);
327                            methods.addAll(getParameterlessMethods(fieldObject, field));
328                    }
329                    return methods;
330        }

332            private static Map<String, ObjectFieldMethod> indexMethods(List<ObjectFieldMethod> methods) throws IllegalArgumentException {
333                    Map<String, ObjectFieldMethod> map = Maps.newHashMap();
334            Set<String> ambiguousNames = Sets.newHashSet();
335            for (ObjectFieldMethod ofm : methods) {
336                    Method method = ofm.getMethod();
337                    String blindKey = method.getName();
338                    if (!ambiguousNames.contains(blindKey)) {
339                            if (map.containsKey(blindKey)) {
340                                    map.remove(blindKey);
341                                    ambiguousNames.add(blindKey);
342                            } else {
343                                    map.put(blindKey, ofm);
344                            }
345                    }
346                    String fieldName = ofm.getField() == null ? "this" : ofm.getField().getName(); 
347                    String qualifiedKey = fieldName + "." + blindKey; 
348                    map.put(qualifiedKey, ofm);
349            }
350            return map;
351            }

353            private List<Field> getBindableModelFields() {
354                    List<Field> allModelFields = Reflections.getFieldsOfTypeForClassHierarchy(object.getClass(), BindableModel.class);
355                    List<Field> notBindableFields = Reflections.getAnnotatedFieldsForClassHierarchy(object.getClass(), NotBindable.class);
356                    allModelFields = ImmutableList.copyOf(Iterables.filter(allModelFields, Predicates.not(Predicates.in(notBindableFields))));
357                    List<Field> nonFinalModelFields = ImmutableList.copyOf(Iterables.filter(allModelFields, Predicates.not(Reflections.IS_FIELD_FINAL)));
358                    if (!nonFinalModelFields.isEmpty()) {
359                            throw new BindingException("All BindableModels have to be final or marked with @NotBindable, but "+
360                                    Iterables.transform(nonFinalModelFields, Reflections.FIELD_TO_NAME)+" are not."); 
361                    }
362                    return allModelFields;
363            }

365            /**
366             * Indexes all bindable models within the binding context. If there are two bindable models
367             * in a class hierarchy with identical names then they are indexed as 
368             * "DeclaringClass.modelFieldName". If this is not unique then one of them will win 
369             * non-deterministically, don't do this.
370             * @return the index
371             */
372            private Map<String, Field> indexBindableModels() {
373                    return dotIndex(getBindableModelFields(), 
374                                    Reflections.FIELD_TO_CONTAINING_CLASS_NAME, 
375                                    Reflections.FIELD_TO_NAME);
376            }
377            
378            /*
379             * TODO Current behavior is if ANY class in a class hierarchy is Bindable then all methods in that
380             * hierarchy are bindable.  Really this should be for each class in the hierarchy, if it's 
381             * marked Bindable then its methods are bindable.
382             */
383            private Map<String, ObjectFieldMethod> indexBindableMethods() throws IllegalArgumentException {
384                    // Get all fields marked @Bindable
385                    List<Field> bindables = getAnnotatedFields(Bindable.class);
386                    if (Iterables.any(bindables, Predicates.not(Reflections.IS_FIELD_FINAL))) {
387                            throw new BindingException("all @Bindables have to be final"); 
388                    }
389                    // Add all BindableModels
390                    bindables.addAll(getBindableModelFields());
391                    
392                    // Index those methods.
393                    List<ObjectFieldMethod> methods = getParameterlessMethodsOnFieldTypes(object, bindables);
394                    
395                    // Add methods for classes marked @Bindable
396            if (Reflections.isClassAnnotatedForClassHierarchy(object, Bindable.class)) {
397                            methods.addAll(Reflections.getParameterlessMethodsForClassHierarchy(object));
398                    }
399            
400            return indexMethods(methods);
401            }
402            
403            private Map<String, ObjectFieldMethod> indexBindableModelMethods() throws IllegalArgumentException {
404                    List<ObjectFieldMethod> methods = getParameterlessMethodsOnFieldTypes(object, getBindableModelFields());
405            return indexMethods(methods);
406            }

408            static <T> boolean isOn(Object onObject, Set<T> changedSet) {
409            if (changedSet.contains(ModelUpdates.ALL)) {
410                    return true;
411            }
412            return changedSet.contains(onObject);
413        }
414    
415            public static <T extends Enum<?> & ModelUpdate> boolean isOn(Object onObject, T... changed) {
416            if (onObject == null) {
417                    return true;
418            }
419            final Set<T> changedSet = Sets.newHashSet(changed);
420            return BindingContext.isOn(onObject, changedSet);
421        }
422    
423            public static <T extends Enum<?> & ModelUpdate> boolean isOn(Collection<Object> ons, T... changed) {
424            if (ons == null || ons.isEmpty()) {
425                    return true;
426            }
427            final Set<T> changedSet = Sets.newHashSet(changed);
428            for (Object on : ons) {
429                    if (BindingContext.isOn(on, changedSet)) {
430                            return true;
431                    }
432            }
433            return false;
434        }

436            public static List<Object> getOnObjects(String[] ons, BindableModel model) {
437            if (ons == null) {
438                    return null;
439            }
440            List<Object> onObjects = Lists.newArrayList();
441            for (int i = 0; i < ons.length; i++) {
442                    Object onObject = findOnObject(ons[i], model);
443                    if (onObject != null) {
444                            onObjects.add(onObject);
445                    }
446            }
447            return onObjects;
448        }
