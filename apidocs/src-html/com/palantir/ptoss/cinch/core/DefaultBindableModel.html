001    package com.palantir.ptoss.cinch.core;
002    
003    import java.util.List;
004    import java.util.concurrent.CopyOnWriteArrayList;
005    
006    public class DefaultBindableModel implements BindableModel {
007    
008            private final List<Binding> bindings = new CopyOnWriteArrayList<Binding>();
009            
010            public void bind(Binding toBind) {
011                    if (!bindings.contains(toBind)) {
012                            bindings.add(toBind);
013                    }
014        }
015    
016            public void unbind(Binding toUnbind) {
017                    bindings.remove(toUnbind);
018        }
019    
020            /**
021             * Removes all bindings from this model.
022             */
023            public void unbindAll() {
024                bindings.clear();
025        }
026            
027            public <T extends Enum<T> & ModelUpdate> void modelUpdated(T... changed) {
028                    for (Binding binding : bindings) {
029                            binding.update(changed);
030                    }
031        }
032            
033            public void update() {
034                    this.modelUpdated(ModelUpdates.UNSPECIFIED);
035            }
036    
037    }
