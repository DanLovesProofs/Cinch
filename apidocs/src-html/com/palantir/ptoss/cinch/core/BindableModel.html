014    package com.palantir.ptoss.cinch.core;
015    
016    /**
017     * A model that can have bindings applied to it.
018     */
019    public interface BindableModel {
020            /**
021             * Attach the binding to the model. Whenever the model is updated then the binding will be
022             * triggered.
023             * @param toBind
024             */
025            void bind(Binding toBind);
026            
027            /**
028             * Removes the passed binding from this model.
029             * @param toUnbind
030             */
031            void unbind(Binding toUnbind);
032            
033            /**
034             * Alert the bindings that the model has updated with the specified change types.
035             * @param changed list of change types to indicate
036             */
037            <T extends Enum<T> & ModelUpdate> void modelUpdated(T... changed);
038    }
