014    package com.palantir.ptoss.cinch.core;
015    
016    import java.util.List;
017    
018    import com.google.common.base.Objects;
019    import com.google.common.collect.Lists;
020    
021    /**
022     * <p>
023     * Superclass for {@link BindableModel} instances, this class handles the binding 
024     * tasks for subclassed models.
025     * </p>
026     * <p>
027     * Basic theory of operation:  
028     * TODO
029     */
030    public class WeakBindableModel implements BindableModel {
031            /**
032             * A delegate object that implements all of the logic behind {@link BindableModel}.
033             */
034            private final WeakBindableModelSupport support = new WeakBindableModelSupport();
035            /**
036             * List of strong bindings.  This {@link List} holds strong references to all
037             * of the {@link Binding} objects passed to this model.
038             */
039            private final List<Binding> strongBindings = Lists.newArrayList();
040    
041            /**
042             * Binds this model to some component with a {@link Binding} object.  This method is called by 
043             * the various {@link WiringHarness} objects to bind individual components to this model. 
044             * 
045             * @param binding
046             */
047            public void bind(Binding binding) {
048                support.bind(binding);
049        }
050            
051            /**
052             * Binds this model to a {@link Binding} using strong references.
053             * 
054             * @param binding
055             */
056            public void bindStrongly(Binding binding) {
057                    support.bind(binding);
058                    strongBindings.add(binding);
059            }
060    
061            /**
062             * {@inheritDoc}
063             */
064            public <T extends Enum<T> & ModelUpdate> void modelUpdated(T... changed) {
065                support.modelUpdated(changed);
066        }
067            
068            /**
069             * Fires a model update if the old and new values are different.
070             * @param oldValue original value to be compared
071             * @param newValue new value to compare against
072             * @param <T> A varargs list of enumerated {@link ModelUpdate} types.
073             */
074            public <T extends Enum<T> & ModelUpdate> void modelUpdated(
075                            Object oldValue, Object newValue, T... changed) {
076                    if (!Objects.equal(oldValue, newValue)) {
077                            modelUpdated(changed);
078                    }
079            }
080            
081            /**
082             * Called by the implementing <code>Model</code> to notify all listeners
083             * that data in the <code>Model</code> has changed.  
084             */
085            public void update() {
086                    support.update();
087            }
088    
089            /**
090             * {@inheritDoc}
091             */
092            public void unbind(Binding binding) {
093                support.unbind(binding);
094                strongBindings.remove(binding);
095        }
096            
097            /**
098             * Unbinds both weak and strong {@link Binding} from this Model.
099             * @see BindableModel#unbind(Binding)
100             */
101            public void unbindAll() {
102                    support.unbindAll();
103                    strongBindings.clear();
104            }
105            
106    }
