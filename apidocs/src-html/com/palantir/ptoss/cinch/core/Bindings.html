016    import java.util.Collection;
017    import java.util.List;
018    
019    import com.google.common.collect.ImmutableList;
020    import com.google.common.collect.Lists;
021    import com.palantir.ptoss.cinch.swing.Action;
022    import com.palantir.ptoss.cinch.swing.Bound;

025    import com.palantir.ptoss.cinch.swing.BoundSelection;

027    import com.palantir.ptoss.cinch.swing.OnChange;
028    import com.palantir.ptoss.cinch.swing.OnClick;
029    import com.palantir.ptoss.cinch.swing.OnFocusChange;
030    import com.palantir.ptoss.cinch.swing.VisibleIf;

032    /**
033     * <p>
034     * All {@link Binding}s are called when {@link ModelUpdates#ALL} is called.
035     * <p>
036     * If a {@link Bound} component has an "on" parameter then it will only be triggered if the
037     * specific {@link ModelUpdate} type is triggered.
038     * <p>
039     * If a {@link Bound} component has no "on" parameter then it will be triggered by any update 
040     * type (including none) triggered by its {@link BindableModel}.
041     */
042    public class Bindings {
043            
044            private final List<Binding> bindings = Lists.newArrayList();
045            
046            private final ImmutableList<BindingWiring> wirings;
047    
048            /**
049             * The list of {@link BindingWiring} classes that are standard to this framework.
050             */
051            public static final ImmutableList<BindingWiring> STANDARD_BINDINGS = ImmutableList.<BindingWiring>builder()
052                    .add(new Bound.Wiring())
053                    .add(new BoundSelection.Wiring())
054                    .add(new BoundExtent.Wiring())
055                    .add(new BoundLocation.Wiring())
056                    .add(new CallOnUpdate.Wiring())
057                    .add(new EnabledIf.Wiring())
058                    .add(new Action.Wiring())
059                    .add(new VisibleIf.Wiring())
060                    .add(new OnClick.Wiring())
061                    .add(new OnChange.Wiring())
062                    .add(new OnFocusChange.Wiring())
063                    .build();
064            
065            public static Bindings standard() {
066                    return new Bindings(STANDARD_BINDINGS);
067            }
068            
069            public Bindings() {
070                    this.wirings = STANDARD_BINDINGS;
071            }
072            
073            public Bindings(Collection<? extends BindingWiring> wirings) {
074                    this.wirings = ImmutableList.copyOf(wirings);
075        }
076            
077            public void bind(Object object) {
078                    bindWithoutUpdate(object);
079                    updateAll();
080            }
081            
082            public void bindWithoutUpdate(Object object) {
083                    BindingContext context = new BindingContext(object);
084                    bindings.addAll(createBindings(context));
085            }
086    
087            public void release(BindableModel model) {
088                    for (Binding binding : bindings) {
089                            model.unbind(binding);
090                    }
091            }
092            
093            protected List<Binding> createBindings(BindingContext context) {
094                    final List<Binding> bindingList = Lists.newArrayList();
095                    for (BindingWiring wiring : wirings) {
096                            bindingList.addAll(wiring.wire(context));
097                    }
098                    return bindingList;
099            }
100    
101            public void updateAll() {
102                    for (Binding binding : bindings) {
103                            binding.update(ModelUpdates.ALL);
104                    }
105            }
106    }
