001    package com.palantir.ptoss.cinch.core;
002    
003    import java.lang.annotation.ElementType;
004    import java.lang.annotation.Retention;
005    import java.lang.annotation.RetentionPolicy;
006    import java.lang.annotation.Target;
007    import java.lang.reflect.InvocationTargetException;
008    import java.lang.reflect.Method;
009    import java.util.Collection;
010    import java.util.List;
011    import java.util.Set;
012    
013    import org.apache.log4j.Logger;
014    
015    import com.google.common.base.Strings;
016    import com.google.common.collect.ImmutableList;
017    import com.google.common.collect.Lists;
018    
019    /**
020     * An annotation to mark that a method should be called when the bound model updates
021     * in a specific way. The method can have any access modifiers, i.e. it can be private.
022     */
023    @Retention(RetentionPolicy.RUNTIME)
024    @Target({ElementType.METHOD})
025    public @interface CallOnUpdate {
026            /**
027             * The model field name to bind to. If this is left blank and the {@link BindingContext} has
028             * only a single {@link ListenableModel} then it will bind to that.
029             */
030            String model() default "";
031    
032            /**
033             * What model update type to trigger on, or blank (default) for all updates.
034             */
035            String[] on() default ""; //$NON-NLS-1$
036    
037            static class Wiring implements BindingWiring {
038                    private static final Logger logger = Logger.getLogger(CallOnUpdate.class);
039    
040                    public Collection<Binding> wire(final BindingContext context) {
041                        final List<ObjectFieldMethod> methods = context.getAnnotatedParameterlessMethods(CallOnUpdate.class);
042                        final List<Binding> bindings = Lists.newArrayList();
043                        for (final ObjectFieldMethod method : methods) {
044                            final CallOnUpdate callOnUpdate = method.getMethod().getAnnotation(CallOnUpdate.class);
045                            bindings.addAll(wire(callOnUpdate, context, method));
046                        }
047                        return bindings;
048                    }
049    
050                    private static Collection<Binding> wire(final CallOnUpdate callOnUpdate, final BindingContext context, final ObjectFieldMethod method) {
051                            final String to = callOnUpdate.model();
052                            final BindableModel model;
053                            if (Strings.isNullOrEmpty(to)) {
054                                    Set<BindableModel> models = context.getBindableModels();
055                                    if (models.size() != 1) {
056                                            throw new BindingException("more than one bindable model for empty 'to'"); //$NON-NLS-1$
057                                    }
058                                    model = models.iterator().next();
059                            } else {
060                            model = context.getBindableModel(to);
061                            if (model == null) {
062                                    throw new BindingException("can't find method to bind to: " + to); //$NON-NLS-1$
063                            }
064                            }
065                            final String[] ons = callOnUpdate.on();
066                            List<Object> onObjects = BindingContext.getOnObjects(ons, model);
067                            Binding binding = makeBinding(method, onObjects);
068                            model.bind(binding);
069                            return ImmutableList.of(binding);
070                    }
071    
072                    private static Binding makeBinding(final ObjectFieldMethod method, final List<Object> onObjects) {
073                            final Method actualMethod = method.getMethod();
074                            actualMethod.setAccessible(true);
075                            final Binding binding = new Binding() {
076                                    public <T extends Enum<?> & ModelUpdate> void update(final T... changed) {
077                                            if (!BindingContext.isOn(onObjects, changed)) {
078                                                    return;
079                                            }
080                                            try {
081                                        actualMethod.invoke(method.getObject());
082                                    } catch (final InvocationTargetException itex) {
083                                            logger.error("exception during CallOnUpdate firing", itex.getCause()); //$NON-NLS-1$
084                                    } catch (final Exception e) {
085                                            logger.error("exception in method binding", e); //$NON-NLS-1$
086                                    }
087                                    }
088                            };
089                            return binding;
090                    }
091            }
092    }
