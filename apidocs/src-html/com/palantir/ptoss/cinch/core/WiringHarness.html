014    package com.palantir.ptoss.cinch.core;
015    
016    import java.beans.IntrospectionException;
017    import java.lang.annotation.Annotation;
018    import java.util.Collection;
019    
020    /**
021     * Interface for classes that actually perform a specific wiring.
022     * @author regs
023     *
024     * @param <T> the type of {@link Annotation} that this harness consumes.
025     * @param <W> the type of component that this harness knows how to wire.
026     */
027    public interface WiringHarness<T extends Annotation, W> {
028            /**
029             * Performs the wiring of a specific component based on the passed binding annotation.
030             * @param bound annotation containing the data needed to wire this component to the context.
031             * @param context metadata about the context for wiring.
032             * @param toWire the component to be wired by this harness.
033             * @return one or more {@link Binding} objects represent the wirings performed by this harness.
034             * @throws IllegalAccessException
035             * @throws IntrospectionException
036             */
037            Collection<? extends Binding> wire(T bound, BindingContext context, W toWire) 
038                    throws IllegalAccessException, IntrospectionException;
039    }
