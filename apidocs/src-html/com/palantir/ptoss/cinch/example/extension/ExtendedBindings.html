014    package com.palantir.ptoss.cinch.example.extension;
015    
016    import java.util.List;
017    
018    import com.google.common.collect.Lists;
019    import com.palantir.ptoss.cinch.core.BindingWiring;
020    import com.palantir.ptoss.cinch.core.Bindings;
021    
022    /**
023     * A sample class that returns extended bindings for including customizations to Cinch.
024     * 
025     * @see Bindings
026     * @see ExtendedExample
027     * @see LoggedModel
028     * 
029     */
030    public class ExtendedBindings {
031            /**
032             * Call this instead of {@link Bindings#standard()} to inject the custom bindings.
033             */
034            public static Bindings extendedBindings() {
035                    // start with standard set of bindings
036                    List<BindingWiring> wirings = Lists.newArrayList(Bindings.STANDARD_BINDINGS);
037                    // add in all additions
038                    wirings.add(new LoggedModel.Wiring());
039                    return new Bindings(wirings);
040            }
041    }
