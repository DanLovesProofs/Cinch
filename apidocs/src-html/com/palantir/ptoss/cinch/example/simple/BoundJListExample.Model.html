014    package com.palantir.ptoss.cinch.example.simple;
015    
016    import java.awt.BorderLayout;
017    import java.awt.EventQueue;
018    import java.lang.reflect.InvocationTargetException;
019    import java.util.List;
020    
021    import javax.swing.*;
022    
023    import com.google.common.base.Joiner;
024    import com.google.common.collect.ImmutableList;
025    import com.palantir.ptoss.cinch.core.Bindable;
026    import com.palantir.ptoss.cinch.core.Bindings;
027    import com.palantir.ptoss.cinch.core.DefaultBindableModel;
028    import com.palantir.ptoss.cinch.example.Examples;
029    import com.palantir.ptoss.cinch.swing.Action;
030    import com.palantir.ptoss.cinch.swing.Bound;
031    import com.palantir.ptoss.cinch.swing.BoundSelection;
032    
033    public class BoundJListExample {
034            
035            @SuppressWarnings("unused")
036            private static class Model extends DefaultBindableModel {
037                    private List<String> items = ImmutableList.of("one", "two", "three", "four");
038                    private List<String> selectedItems = ImmutableList.of();
039                    
040                    public List<String> getItems() {
041                    return items;
042            }
043                    
044                    public void setItems(List<String> items) {
045                    this.items = items;
046                    update();
047            }
048                    
049                    public void setSelectedItems(List<String> selectedItems) {
050                            if (selectedItems == null) {
051                                    selectedItems = ImmutableList.of();
052                            }
053                    this.selectedItems = ImmutableList.copyOf(selectedItems);
054                    update();
055            }
056                    
057                    public List<String> getSelectedItems() {
058                    return selectedItems;
059            }
060                    
061                    public String getSelectedString() {
062                            return "selected: " + Joiner.on(",").join(getSelectedItems());
063                    }
064            }
065            
066            @SuppressWarnings("unused")
067            private static class Controller {
068                    private final Model model;
069    
070                    public Controller(Model model) {
071                    this.model = model;
072            }
073                    
074                    public void changeItems() {
075                            if (model.getItems().contains("one")) {
076                                    model.setItems(ImmutableList.of("two", "three", "four", "five", "six"));
077                            } else {
078                                    model.setItems(ImmutableList.of("one", "two", "three", "four"));
079                            }
080                    }
081            }
082            
083            private final Model model = new Model();
084            
085            @SuppressWarnings("unused")
086            @Bindable
087            private final Controller controller = new Controller(model);
088            
089            @Bound(to = "items")
090            @BoundSelection(to = "selectedItems", multi = true)
091            private final JList list = new JList();
092            
093            private final JPanel panel = new JPanel();
094            
095            @Action(call = "changeItems")
096            private final JButton changeItems = new JButton("Change");
097    
098            @Bound(to = "selectedString")
099            private final JLabel selectedLabel = new JLabel(" ");
100            
101            private final Bindings bindings = Bindings.standard();
102            
103            public BoundJListExample() {
104                    initializeInterface();
105                    bindings.bind(this);
106        }
107            
108            private void initializeInterface() {
109                    panel.setLayout(new BorderLayout());
110                    panel.add(selectedLabel, BorderLayout.NORTH);
111                    panel.add(list, BorderLayout.CENTER);
112                    panel.add(changeItems, BorderLayout.SOUTH);
113                    
114                    JFrame frame = new JFrame("Demo"); 
115                    frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
116                    frame.setLocation(100, 100);
117                    frame.setSize(400, 600);
118            
119                    frame.setContentPane(panel);
120                    
121                    frame.setVisible(true);
122            }
123            
124            public static void main(String[] args) throws InterruptedException, InvocationTargetException {
125                    Examples.initializeLogging();
126                    EventQueue.invokeAndWait(new Runnable() {
127                    public void run() {
128                            new BoundJListExample();
129                    }
130            });
131        }
132    }
