016    import java.awt.*;
017    import java.lang.reflect.InvocationTargetException;
018    
019    import javax.swing.*;
020    import javax.swing.plaf.metal.MetalSliderUI;
021    
022    import org.apache.log4j.LogManager;
023    import org.apache.log4j.Logger;
024    
025    import com.palantir.ptoss.cinch.core.Bindings;

031    /**
032     * An illustration of using the {@link BoundExtent} annotation to tie the extent of one control to
033     * another. In this example, the top slider in the UI sets the extent for the bottom slider.
034     * @see JSlider#setExtent(int)
035     * @see BoundExtent
036     */
037    public class BoundExtentsExample {
038    
039            static final Logger log = LogManager.getLogger(BoundExtentsExample.class);
040    
041            public static class Model extends DefaultBindableModel {
042                    private int sliderExtent;
043    
044                    public int getSliderExtent() {
045                            return sliderExtent;
046                    }
047    
048                    public void setSliderExtent(int sliderPosition) {
049                            this.sliderExtent = sliderPosition;
050                            log.debug("Slider Extent set: " + sliderPosition);
051                            update();
052                    }
053    
054            }
055    
056            @SuppressWarnings("unused")
057            private final Model model = new Model();
058    
059            private final JPanel panel = new JPanel();
060    
061            @BoundExtent(to = "sliderExtent")
062            private final JSlider slider = new JSlider(new DefaultBoundedRangeModel(0, 0, 0, 100));
063    
064            @Bound(to = "sliderExtent")
065            private final JSlider controlSlider = new JSlider(new DefaultBoundedRangeModel(0, 0, 0, 100));
066    
067            @Bound(to = "sliderExtent")
068            private final JLabel extent = new JLabel();
069            private final Bindings bindings = Bindings.standard();
070    
071            public BoundExtentsExample() {
072                    initializeInterface();
073                    bindings.bind(this);
074            }
075    
076            private JPanel buildTextPanel() {
077                    JPanel panel = new JPanel(new BorderLayout());
078                    panel.add(extent, BorderLayout.CENTER);
079                    panel.add(new JLabel("Extent: "), BorderLayout.WEST);
080                    return panel;
081            }
082    
083            private void styleSlider(JSlider slider) {
084                    slider.setMajorTickSpacing(10);
085                    slider.setMinorTickSpacing(1);
086                    slider.setPaintLabels(true);
087                    slider.setPaintTicks(true);
088                    slider.setPaintTrack(true);
089                    slider.setSnapToTicks(true);
090            }
091    
092            private void addExtentsHighlight(JSlider slider) {
093                    slider.setUI(new ExtentHighlightingSliderUI(slider));
094            }
095    
096            private void initializeInterface() {
097                    panel.setLayout(new BorderLayout());
098                    styleSlider(slider);
099                    addExtentsHighlight(slider);
100                    panel.add(slider, BorderLayout.CENTER);
101                    styleSlider(controlSlider);
102                    panel.add(controlSlider, BorderLayout.NORTH);
103                    panel.add(buildTextPanel(), BorderLayout.SOUTH);
104    
105                    JFrame frame = new JFrame(getClass().getSimpleName());
106                    frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
107                    frame.setLocation(100, 100);
108                    frame.setSize(400, 200);
109    
110                    frame.setContentPane(panel);
111    
112                    frame.setVisible(true);
113            }
114    
115            public static void main(String[] args) throws InterruptedException, InvocationTargetException {
116                    Examples.initializeLogging();
117                    // set to Metal LaF
118                    try {
119                            UIManager.setLookAndFeel(UIManager.getCrossPlatformLookAndFeelClassName());
120                            EventQueue.invokeAndWait(new Runnable() {
121                                    public void run() {
122                                            new BoundExtentsExample();
123                                    }
124                            });
125                    } catch (Exception e) {
126                            e.printStackTrace();
127                    }
128            }
129    
130            static class ExtentHighlightingSliderUI extends MetalSliderUI {
131    
132                    final JSlider slider;
133                    Color originalLabelColor = null;
134                    static final Color RED = Color.RED;
135    
136                    public ExtentHighlightingSliderUI(JSlider slider) {
137                            this.slider = slider;
138                    }
139    
140                    public void setOriginalLabelColor(Color originalLabelColor) {
141                            if (originalLabelColor == null || !RED.equals(originalLabelColor)) {
142                                    this.originalLabelColor = originalLabelColor;
143                            }
144                    }
145    
146                    protected void paintHorizontalLabel(Graphics g, int value, Component label) {
147                            final int extentLowerBound = slider.getMaximum() - slider.getExtent();
148                            setOriginalLabelColor(label.getForeground());
149                            if (value > extentLowerBound) {
150                                    label.setForeground(Color.RED);
151                            } else {
152                                    label.setForeground(originalLabelColor);
153                            }
154                            super.paintHorizontalLabel(g, value, label);
155                    }
156    
157                    protected void paintMinorTickForHorizSlider(Graphics g, Rectangle tickBounds, int x) {
158                            super.paintMinorTickForHorizSlider(g, tickBounds, x);
159                            Color original = g.getColor();
160                            if (valueForXPosition(x) > slider.getMaximum() - slider.getExtent()) {
161                                    g.setColor(Color.RED);
162                            }
163                            g.drawLine(x, TICK_BUFFER, x, TICK_BUFFER + (tickLength / 2));
164                            g.setColor(original);
165                    }
166    
167                    protected void paintMajorTickForHorizSlider(Graphics g, Rectangle tickBounds, int x) {
168                            super.paintMajorTickForHorizSlider(g, tickBounds, x);
169                            Color original = g.getColor();
170                            if (valueForXPosition(x) > slider.getMaximum() - slider.getExtent()) {
171                                    g.setColor(Color.RED);
172                            }
173                            g.drawLine(x, TICK_BUFFER, x, TICK_BUFFER + (tickLength - 1));
174                            g.setColor(original);
175                    }
176            }
177    }
