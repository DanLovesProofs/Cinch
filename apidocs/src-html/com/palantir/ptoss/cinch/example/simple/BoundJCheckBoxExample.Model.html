016    import java.awt.BorderLayout;
017    import java.awt.EventQueue;
018    import java.lang.reflect.InvocationTargetException;
019    
020    import javax.swing.JButton;
021    import javax.swing.JCheckBox;
022    import javax.swing.JFrame;
023    import javax.swing.JLabel;
024    import javax.swing.JPanel;
025    
026    import com.palantir.ptoss.cinch.core.Bindable;
027    import com.palantir.ptoss.cinch.core.Bindings;
028    import com.palantir.ptoss.cinch.core.DefaultBindableModel;
029    import com.palantir.ptoss.cinch.example.Examples;
030    import com.palantir.ptoss.cinch.swing.Action;
031    import com.palantir.ptoss.cinch.swing.Bound;

033    /**
034     * <p>
035     * This example shows how controls can be bound to methods on a controller object, 
036     * <code>controller</code>.
037     * </p>
038     * <p>
039     * Note that <code>controller</code> must be marked as {@link Bindable} to enable this behavior.
040     * </p>
041     * @see Action
042     * @see Bindable
043     * @see Bound
044     */
045    public class BoundJCheckBoxExample {
046            public static class Model extends DefaultBindableModel {
047                    private boolean state;
048                    
049                    public void setState(boolean state) {
050                    this.state = state;
051                    update();
052            }
053                    
054                    public boolean isState() {
055                    return state;
056            }
057            }
058            
059            public static class Controller {
060                    private final Model model;
061    
062                    public Controller(Model model) {
063                    this.model = model;
064            }
065                    
066                    public void setToTrue() {
067                            model.setState(true);
068                    }
069                    
070                    public void setToFalse() {
071                            model.setState(false);
072                    }
073            }
074            
075            private final Model model = new Model();
076            @SuppressWarnings("unused")
077            @Bindable
078            private final Controller controller = new Controller(model);
079            
080            @Bound(to = "state")
081            private final JCheckBox box = new JCheckBox("State");
082            @Bound(to = "state")
083            private final JLabel stateLabel = new JLabel("?");
084            
085            @Action(call = "setToTrue")
086            private final JButton trueButton = new JButton("Set True");
087            @Action(call = "setToFalse")
088            private final JButton falseButton = new JButton("Set False");
089            
090            private final JPanel panel = new JPanel();
091            
092            private final Bindings bindings = Bindings.standard();
093            
094            public BoundJCheckBoxExample() {
095                    panel.setLayout(new BorderLayout());
096                    panel.add(box, BorderLayout.CENTER);
097                    panel.add(stateLabel, BorderLayout.SOUTH);
098                    JPanel buttons = new JPanel();
099                    buttons.add(trueButton);
100                    buttons.add(falseButton);
101                    panel.add(buttons, BorderLayout.NORTH);
102                    bindings.bind(this);
103        }
104            
105            public static void main(String[] args) throws InterruptedException, InvocationTargetException {
106                    EventQueue.invokeAndWait(new Runnable() {
107                    public void run() {
108                            BoundJCheckBoxExample example = new BoundJCheckBoxExample();
109                            JFrame frame = Examples.getFrameFor("Cinch JCheckBox Example", example.panel);
110                            frame.pack();
111                                    frame.setVisible(true);
112                    }
113            });
114        }
115    }
