014    package com.palantir.ptoss.cinch.example.demo;
015    
016    import java.util.List;
017    
018    import com.google.common.base.Predicate;
019    import com.google.common.collect.ImmutableList;
020    import com.google.common.collect.Iterables;
021    import com.google.common.collect.Lists;
022    import com.palantir.ptoss.cinch.core.DefaultBindableModel;
023    import com.palantir.ptoss.cinch.core.ModelUpdate;
024    
025    
026    public class DemoModel extends DefaultBindableModel {
027            
028            public enum UpdateType implements ModelUpdate {
029                    OTHER, LIST, COMBO, MULTILIST;
030            }
031    
032            public enum DemoEnum {
033                    FOO, BAR, BAZ;
034            }
035            
036            private boolean demoBoolean;
037            private boolean demoRadioBoolean;
038            private DemoEnum demoEnum;
039            private String demoString = ""; 
040            private List<String> demoList = ImmutableList.<String>of("Quick", "Quickly", "Quickest", "Brown", "Fox");
041            private List<String> demoFilteredList = demoList;
042            private Object selectedItem;
043            private String filterText;
044            private String selectedComboItem;
045            private Object selectedPrefItem;
046            
047            private List<String> demoMultiList = ImmutableList.<String>of("alpha", "bravo", "charlie", "delta", "echo"); 
048            private List<String> multiSelectedItems = ImmutableList.of();
049            
050            private int sliderValue = 0;
051            
052            private final Predicate<String> filter = new Predicate<String>() {
053                    public boolean apply(String input) {
054                            if (filterText == null || filterText.trim().length() == 0) {
055                                    return true;
056                            }
057                            return input.toLowerCase().startsWith(filterText.trim().toLowerCase());
058                    }
059            };
060            
061            public void setDemoList(List<String> objects) {
062                    this.demoList = objects;
063                    applyFilter();          
064                    modelUpdated(UpdateType.LIST);
065            }
066            
067            public List<String> getDemoList() {
068                return demoList;
069        }
070    
071            public void setSelectedItem(Object item) {
072                    selectedItem = item;
073                    update();
074            }
075            
076            public Object getSelectedItem() {
077                return selectedItem;
078            }
079            
080            public List<DemoEnum> getDemoEnumValues() {
081                    return ImmutableList.copyOf(DemoEnum.values());
082            }
083    
084            public Object getSelectedPrefItem() {
085                    return selectedPrefItem;
086            }
087    
088            public void setSelectedPrefItem(Object selectedPrefItem) {
089                    this.selectedPrefItem = selectedPrefItem;
090                    update();
091            }
092    
093            public void setDemoBoolean(boolean demoBoolean) {
094                    this.demoBoolean = demoBoolean;
095                    update();
096            }
097            
098            public boolean isDemoBoolean() {
099                    return demoBoolean;
100            }
101            
102            public void setDemoEnum(DemoEnum demoEnum) {
103                this.demoEnum = demoEnum;
104                update();
105        }
106            
107            public DemoEnum getDemoEnum() {
108                return demoEnum;
109        }
110    
111            public void setDemoString(String demoString) {
112                this.demoString = demoString;
113                update();
114        }
115            
116            public String getDemoString() {
117                return demoString;
118        }
119            
120            public List<String> getComboList() {
121                    return ImmutableList.of("Alpha", "Bravo", "Charlie");
122            }
123            
124            private void applyFilter() {
125                    demoFilteredList = Lists.newArrayList(Iterables.filter(demoList, filter));
126            }
127    
128            public List<String> getFilteredList() {
129                    return demoFilteredList;
130        }
131            
132            public void setFilterText(String filterText) {
133                this.filterText = filterText;
134                applyFilter();
135                modelUpdated(UpdateType.LIST);
136        }
137            
138            public String getFilterText() {
139                return filterText;
140        }
141            
142            public void setSelectedComboItem(String selectedComboItem) {
143                this.selectedComboItem = selectedComboItem;
144                update();
145        }
146            
147            public String getSelectedComboItem() {
148                return selectedComboItem;
149        }
150            
151            public String getSelectedComboItemString() {
152                    return "selection: " + selectedComboItem;
153            }
154            
155            public void setDemoRadioBoolean(boolean demoRadioBoolean) {
156                this.demoRadioBoolean = demoRadioBoolean;
157                update();
158        }
159            
160            public boolean isDemoRadioBoolean() {
161                return demoRadioBoolean;
162        }
163            
164            public List<String> getDemoMultiList() {
165                return demoMultiList;
166        }
167            
168            public void setDemoMultiList(List<String> demoMultiList) {
169                this.demoMultiList = demoMultiList;
170                modelUpdated(UpdateType.MULTILIST);
171            }
172            
173            public List<String> getMultiSelectedItems() {
174                return multiSelectedItems;
175        }
176            
177            public void setMultiSelectedItems(List<String> multiSelectedItems) {
178                this.multiSelectedItems = multiSelectedItems;
179                update();
180        }
181            
182            public void setSliderValue(int sliderValue) {
183                this.sliderValue = sliderValue;
184                update();
185        }
186            
187            public int getSliderValue() {
188                return sliderValue;
189        }
190    }
