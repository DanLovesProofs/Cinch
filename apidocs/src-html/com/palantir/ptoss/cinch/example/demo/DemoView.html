135            public static DemoView create() {
136                    DemoView demo = new DemoView();
137                    demo.bindings.bind(demo);
138                    demo.showUi();
139                    return demo;
140            }
141            
142            private static JPanel panelOf(JComponent... comps) {
143                    JPanel p = new JPanel();
144                    p.setOpaque(false);
145                    for (JComponent comp : comps) {
146                            p.add(comp);
147                    }
148                    return p;
149            }
150            
151            private void showUi() {
152                    JFrame frame = new JFrame("Demo"); 
153                    frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
154                    frame.setLocation(100, 100);
155                    frame.setSize(600, 800);
156                    
157                    JPanel controlsPanel = new JPanel();
158                    controlsPanel.setOpaque(false);
159                    BoxLayout box = new BoxLayout(controlsPanel, BoxLayout.Y_AXIS);
160                    controlsPanel.setLayout(box);
161                    controlsPanel.add(panelOf(
162                                    new JLabel("Enabling Button:"),
163                            demoBooleanCheckBox,
164                            demoButton1,
165                            demoButton2));
166                    controlsPanel.add(panelOf(
167                                    new JLabel("Bind Radio Buttons to boolean:"),
168                            trueButton,
169                            falseButton));
170                    controlsPanel.add(new JSeparator());
171                    controlsPanel.add(panelOf(
172                                    new JLabel("Bind Radio Buttons to enum:"),
173                            fooButton,
174                            barButton,
175                            bazButton,
176                            setToFooButton));
177                    controlsPanel.add(new JSeparator());
178                    controlsPanel.add(panelOf(
179                                    new JLabel("Filtering JList:"),
180                            filterText,
181                            demoList,
182                            changeList,
183                            changeSelection,
184                            listChangedLabel));
185                    controlsPanel.add(new JSeparator());
186                    controlsPanel.add(panelOf(
187                                    new JLabel("Multiselect JList:"),
188                                    demoMultiselectList,
189                                    multiselectContents,
190                                    selectMulti));
191                    controlsPanel.add(new JSeparator());
192                    controlsPanel.add(panelOf(
193                                    new JLabel("Bind JComboBox:"),
194                            comboBox,
195                            selItem));
196                    controlsPanel.add(new JSeparator());
197                    controlsPanel.add(panelOf(
198                                    new JLabel("Slider:"),
199                                    slider,
200                                    sliderValue,
201                                    setSlider));
202                    controlsPanel.add(new JSeparator());
203                    
204                    String text = "test\nlabel\ttab";
205                    JLabel testLabel = new JLabel(text);
206                    JTextField testField = new JTextField(text);
207                    controlsPanel.add(panelOf(
208                                    testLabel,
209                                    testField));
210                                    
211                    controlsPanel.add(panelOf(
212                                    new JLabel("Custom component:"),
213                            pointButton,
214                            lineButton,
215                            allowDrag,
216                            clearButton));
217                    controlsPanel.add(canvas);
218                    
219                    panel.setLayout(new BorderLayout());
220                    panel.add(controlsPanel, BorderLayout.NORTH);
221                    panel.add(canvas, BorderLayout.CENTER);
222                    frame.setContentPane(panel);
223                    frame.setVisible(true);
224            }

227                    System.out.println("listClicked");
228            }
229            
230            @CallOnUpdate(model = "demoModel")
231            public void updateBackgroundColor() {
232                    Color color = null;
233                    if (demoModel.getDemoEnum() == DemoEnum.BAZ) {
234                            color = Color.RED;
235                    }
236                    panel.setBackground(color);
237                    panel.repaint();
238            }
239            
240            int listChangeCount = 0;
241            @CallOnUpdate(model = "demoModel", on = "LIST")
242            public void listChanged() {
243                    ++listChangeCount;
244                    listChangedLabel.setText("list changes: " + listChangeCount); 
245            }
246    
247            public void setSlider() {
248                    demoModel.setSliderValue(50);
249            }
