014    package com.palantir.ptoss.cinch.example.superdemo;
015    
016    import java.awt.EventQueue;
017    
018    import javax.swing.JButton;
019    import javax.swing.JCheckBox;
020    import javax.swing.JFrame;
021    import javax.swing.JPanel;
022    
023    import com.palantir.ptoss.cinch.core.BindableModel;
024    import com.palantir.ptoss.cinch.core.Bindings;
025    import com.palantir.ptoss.cinch.swing.Bound;
026    import com.palantir.ptoss.cinch.swing.EnabledIf;
027    
028    /**
029     * This example shows off the following features:
030     * <ol>
031     * <li>Binding a control to a model field: <code>box</code> is bound to the <code>someBoolean</code>
032     * field on <code>model</code>.</li>
033     * <li>The {@link EnabledIf} annotation: <code>button</code> is only enabled when 
034     * <code>someBoolean</code> is set to <em>true</em>.</li>
035     * <li>The implicit model binding: <code>model</code> is implicitly bound since it's an instance of
036     * {@link BindableModel}.</li>
037     * </ol>
038     * @author regs
039     *
040     */
041    public class SuperDemo {
042            
043            protected final JPanel panel = new JPanel();
044            @Bound(to = "someBoolean")
045            private final JCheckBox box = new JCheckBox("Box");
046            @EnabledIf(to = "someBoolean")
047            private final JButton button = new JButton("Button");
048            private final Bindings bindings = new Bindings();
049    
050            @SuppressWarnings("unused") // binding
051        private final SuperDemoModel model = new SuperDemoModel();
052            
053            protected SuperDemo() {
054                    setupPanel();
055                    bindings.bind(this);
056            }
057            
058            protected void setupPanel() { 
059    //              panel.setLayout(new MigLayout());
060                    panel.add(box);
061                    panel.add(button, "wrap");
062            }
063            
064            protected void showUi() {
065                    JFrame frame = new JFrame("Demo");
066                    frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
067                    frame.setLocation(100, 100);
068                    frame.setSize(600, 600);
069                    
070                    frame.setContentPane(panel);
071                    frame.setVisible(true);
072            }
073            
074            public static SuperDemo create() {
075                    SuperDemo demo = new SuperDemo();
076                    demo.showUi();
077                    return demo;
078            }
079            
080            public static void main(String[] args) {
081                    EventQueue.invokeLater(new Runnable() {
082                            public void run() {
083                                    SuperDemo.create();
084                            }
085                    });
086            }
087    }
