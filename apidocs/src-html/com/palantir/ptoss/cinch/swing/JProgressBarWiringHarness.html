001    package com.palantir.ptoss.cinch.swing;
002    
003    import java.beans.IntrospectionException;
004    import java.lang.reflect.Field;
005    import java.lang.reflect.Method;
006    import java.util.Collection;
007    import java.util.Collections;
008    
009    import javax.swing.JProgressBar;
010    
011    import com.palantir.ptoss.cinch.core.*;
012    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
013    
014    /**
015     * A {@link WiringHarness} for binding a {@link JProgressBar} to an <code>int</code> value in a {@link BindableModel}.
016     */
017    public class JProgressBarWiringHarness implements WiringHarness<Bound, Field> {
018                    public Collection<Binding> wire(Bound bound, BindingContext context, Field field) throws IllegalAccessException, IntrospectionException {
019                            String target = bound.to();
020                    JProgressBar bar = context.getFieldObject(field, JProgressBar.class); 
021    //            ObjectFieldMethod setter = context.findSetter(target);
022                ObjectFieldMethod getter = context.findGetter(target);
023                if (getter == null) {
024                    throw new IllegalArgumentException("could not find getter/setter for " + target); 
025                }
026                BindableModel model = context.getFieldObject(getter.getField(), BindableModel.class);
027                // verify type parameters
028                return bindJProgressBar(model, bar, getter.getMethod());
029            }
030                    
031                    private static int getValueForObject(Object obj) {
032                            if (obj instanceof Double) {
033                                    return (int)Math.round(((Double)obj).doubleValue() * 100);
034                            } else if (obj instanceof Integer) {
035                                    return ((Integer)obj).intValue();
036                            } else if (obj instanceof Float) {
037                                    return (int)Math.round(((Float)obj).doubleValue() * 100);
038                            } 
039                            return -1;
040                    }
041                    
042                    public static Collection<Binding> bindJProgressBar(final BindableModel model, final JProgressBar bar, 
043                                    final Method getter) {
044                        Binding binding = new Binding() {
045                            public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
046                                            try {
047                                                    int val = getValueForObject(getter.invoke(model));
048                                                    bar.setValue(val);
049                                                    bar.setIndeterminate(val < 0);
050                                            } catch (Exception ex) {
051                                                    Wiring.logger.error("exception in JTextField binding", ex); 
052                                            }
053                            }
054                        };
055                        model.bind(binding);
056                        return Collections.singleton(binding);
057                    }
058            }
