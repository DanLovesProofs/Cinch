001    package com.palantir.ptoss.cinch.swing;
002    
003    import java.beans.IntrospectionException;
004    import java.lang.reflect.Field;
005    import java.util.Collection;
006    
007    import javax.swing.event.DocumentEvent;
008    import javax.swing.event.DocumentListener;
009    import javax.swing.text.JTextComponent;
010    
011    import com.google.common.collect.ImmutableList;
012    import com.palantir.ptoss.cinch.core.*;
013    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
014    import com.palantir.ptoss.util.Mutator;
015    
016    /**
017     * A {@link WiringHarness} for binding a {@link JTextComponent} to a {@link String} value in a 
018     * {@link BindableModel}.
019     */
020    public class JTextComponentWiringHarness implements WiringHarness<Bound, Field> {
021            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) throws IllegalAccessException, IntrospectionException {
022                    JTextComponent textComponent = context.getFieldObject(field, JTextComponent.class);
023                    Mutator mutator = Mutator.create(context, bound.to());
024                    Binding binding = bindJTextComponent(mutator, textComponent);
025                    if (binding == null) {
026                            return ImmutableList.of();
027                    }
028                    return ImmutableList.of(binding);
029            }
030    
031            public static Binding bindJTextComponent(final Mutator mutator, final JTextComponent textField) { 
032                    if (mutator.getSetter() != null) {
033                            textField.getDocument().addDocumentListener(new DocumentListener() {
034                                    public void removeUpdate(DocumentEvent e) {
035                                            updateModel();
036                                    }
037    
038                                    public void insertUpdate(DocumentEvent e) {
039                                            updateModel();
040                                    }
041    
042                                    public void changedUpdate(DocumentEvent e) {
043                                            updateModel();
044                                    }
045    
046                                    private void updateModel() {
047                                            try {
048                                                    mutator.set(textField.getText());
049                                            } catch (Exception ex) {
050                                                    Wiring.logger.error("exception in JTextField binding", ex); 
051                                            }                       
052                                    }
053                            });
054                    }
055                    Binding binding = null;
056                    if (mutator.getGetter() != null) {
057                            binding = new Binding() {
058                                    public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
059                                            try {
060                                                    String string = (String)mutator.get();
061                                                    if (string == null) {
062                                                            string = ""; 
063                                                    }
064                                                    if (!string.equals(textField.getText())) {
065                                                            textField.setText(string);
066                                                    }
067                                            } catch (Exception ex) {
068                                                    Wiring.logger.error("exception in JTextField binding", ex); 
069                                            }
070                                    }
071                            };
072                            mutator.getModel().bind(binding);
073                    }
074                    return binding;
075        }
076    }
