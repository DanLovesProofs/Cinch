014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.beans.IntrospectionException;
017    import java.lang.annotation.ElementType;
018    import java.lang.annotation.Retention;
019    import java.lang.annotation.RetentionPolicy;
020    import java.lang.annotation.Target;
021    import java.lang.reflect.Field;
022    import java.lang.reflect.Method;
023    import java.util.Collection;
024    import java.util.Collections;
025    import java.util.List;
026    
027    import org.apache.log4j.Logger;
028    
029    import com.google.common.collect.Lists;
030    import com.palantir.ptoss.cinch.core.*;
031    
032    /**
033     * A binding that will set the enabled state of the annotated component to the state of a model
034     * boolean. The component must have a "setEnabled" method that takes a boolean.
035     */
036    @Retention(RetentionPolicy.RUNTIME)
037    @Target({ElementType.FIELD})
038    public @interface EnabledIf {
039            /**
040             * Enum to specify if normal or inverted comparisons should be used.
041             */
042            public enum Type { NORMAL, INVERTED };
043            /**
044             * The model boolean property to bind to.
045             */
046            String to();
047    
048            /**
049             * Whether or not to invert the boolean.
050             */
051            Type type() default Type.NORMAL;
052    
053            /**
054             * Inner utility class that performs the runtime wiring of all {@link EnabledIf} bindings.
055             * 
056             * @see Bindings#STANDARD_BINDINGS
057             */
058            public static class Wiring implements BindingWiring {
059                    private static final Logger logger = Logger.getLogger(EnabledIf.class);
060    
061                    public Collection<Binding> wire(final BindingContext context) {
062                            final List<Field> actions = context.getAnnotatedFields(EnabledIf.class);
063                            final List<Binding> bindings = Lists.newArrayList();
064                            for (final Field field : actions) {
065                                    final EnabledIf action = field.getAnnotation(EnabledIf.class);
066                                    final String to = action.to();
067                                    final boolean invert = (action.type() == Type.INVERTED);
068                                    try {
069                                            bindings.addAll(wire(to, field, context, invert));
070                                    } catch (final Exception e) {
071                                            throw new BindingException("could not wire up @EnabledIf on " + field.getName(), e); 
072                                    }
073                            }
074                            return bindings;
075                    }
076    
077                    private static Collection<Binding> wire(final String to, final Field field, final BindingContext context, final boolean invert)
078                            throws SecurityException, NoSuchMethodException, IllegalArgumentException, IntrospectionException {
079                            final Method setEnabledMethod = field.getType().getMethod("setEnabled", boolean.class); 
080                            if (setEnabledMethod == null) {
081                                    throw new BindingException("no setEnabled call on EnabledIf field: " + field); 
082                            }
083                            final Object setEnabledObject = context.getFieldObject(field, Object.class);
084                            final ObjectFieldMethod getter = context.findGetter(to);
085                            if (getter == null) {
086                                    throw new BindingException("could not find bindable property: " + to); 
087                            }
088                            if (getter.getMethod().getReturnType() != boolean.class) {
089                                    throw new BindingException("EnabledIf binding must return boolean: " + to); 
090                            }
091                            getter.getMethod().setAccessible(true);
092                            final Binding binding = new Binding() {
093                                    public <T extends Enum<?> & ModelUpdate> void update(final T... changed) {
094                                            try {
095                                                    boolean enabled = (Boolean)getter.getMethod().invoke(getter.getObject());
096                                                    if (invert) {
097                                                            enabled = !enabled;
098                                                    }
099                                                    setEnabledMethod.invoke(setEnabledObject, enabled);
100                                            } catch (final Exception e) {
101                                                    Wiring.logger.error("exception during EnabledIf binding", e); 
102                                            }
103                                    }
104                            };
105                            ((BindableModel)getter.getObject()).bind(binding);
106                            return Collections.singleton(binding);
107                    }
108            }
109    }
