014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.beans.IntrospectionException;
017    import java.lang.reflect.Field;
018    import java.lang.reflect.Method;
019    import java.util.Collection;
020    import java.util.Collections;
021    import java.util.List;
022    
023    import javax.swing.DefaultComboBoxModel;
024    import javax.swing.JComboBox;
025    
026    import com.palantir.ptoss.cinch.core.*;
027    import com.palantir.ptoss.cinch.swing.Bound.Utilities;
028    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
029    
030    /**
031     * A {@link WiringHarness} for binding a {@link JComboBox} to a {@link List} value in a 
032     * {@link BindableModel}.
033     */
034    public class JComboBoxWiringHarness implements WiringHarness<Bound, Field> {
035            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) throws IllegalAccessException, IntrospectionException {
036                    String target = bound.to();
037                    JComboBox combo = context.getFieldObject(field, JComboBox.class);
038                    ObjectFieldMethod getter = context.findGetter(target);
039            BindableModel model = context.getFieldObject(getter.getField(), BindableModel.class);
040            // verify type parameters
041            return bindJComboBox(bound, context, model, combo, getter.getMethod());
042            }
043    
044            private Collection<Binding> bindJComboBox(final Bound bound, final BindingContext context, final BindableModel model, final JComboBox combo, final Method getter) {
045                    final List<Object> ons = BindingContext.getOnObjects(bound.on(), model);
046                    final String nullValue = (String)Utilities.getNullValue(context, bound.nullValue());
047            Binding binding = new Binding() {
048                    public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
049                            if (!BindingContext.isOn(ons, changed)) {
050                                    return;
051                            }
052                                    try {
053                        updateComboModel(combo, (List<?>)getter.invoke(model), nullValue);
054                                    } catch (Exception ex) {
055                            Wiring.logger.error("exception in JList binding", ex); 
056                    }
057                            }
058                    };
059                    model.bind(binding);
060            return Collections.singleton(binding);
061        }
062            
063            private void updateComboModel(JComboBox combo, List<?> newContents, String nullValue) {
064                    int selectedIndex = combo.getSelectedIndex();
065                    Object selected = combo.getSelectedItem();
066                    DefaultComboBoxModel comboModel = new DefaultComboBoxModel();
067                    if (nullValue != null) {
068                            comboModel.addElement(nullValue);
069                    }
070                    for (Object obj : newContents) {
071                            comboModel.addElement(obj);
072                    }
073                    if (comboModel.getIndexOf(selected) != -1) {
074                            comboModel.setSelectedItem(selected);
075                    } else {
076                            if (comboModel.getSize() > selectedIndex) {
077                                    comboModel.setSelectedItem(combo.getItemAt(selectedIndex));
078                            } else if (comboModel.getSize() > 0) {
079                                    comboModel.setSelectedItem(combo.getItemAt(comboModel.getSize() - 1));
080                            }
081                    }
082                    combo.setModel(comboModel);
083            }
084    }
