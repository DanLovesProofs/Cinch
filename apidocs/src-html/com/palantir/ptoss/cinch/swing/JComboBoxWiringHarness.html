001    package com.palantir.ptoss.cinch.swing;
002    
003    import java.beans.IntrospectionException;
004    import java.lang.reflect.Field;
005    import java.lang.reflect.Method;
006    import java.util.Collection;
007    import java.util.Collections;
008    import java.util.List;
009    
010    import javax.swing.DefaultComboBoxModel;
011    import javax.swing.JComboBox;
012    
013    import com.palantir.ptoss.cinch.core.*;
014    import com.palantir.ptoss.cinch.swing.Bound.Utilities;
015    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
016    
017    /**
018     * A {@link WiringHarness} for binding a {@link JComboBox} to a {@link List} value in a 
019     * {@link BindableModel}.
020     */
021    public class JComboBoxWiringHarness implements WiringHarness<Bound, Field> {
022            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) throws IllegalAccessException, IntrospectionException {
023                    String target = bound.to();
024                    JComboBox combo = context.getFieldObject(field, JComboBox.class);
025                    ObjectFieldMethod getter = context.findGetter(target);
026            BindableModel model = context.getFieldObject(getter.getField(), BindableModel.class);
027            // verify type parameters
028            return bindJComboBox(bound, context, model, combo, getter.getMethod());
029            }
030    
031            private Collection<Binding> bindJComboBox(final Bound bound, final BindingContext context, final BindableModel model, final JComboBox combo, final Method getter) {
032                    final List<Object> ons = BindingContext.getOnObjects(bound.on(), model);
033                    final String nullValue = (String)Utilities.getNullValue(context, bound.nullValue());
034            Binding binding = new Binding() {
035                    public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
036                            if (!BindingContext.isOn(ons, changed)) {
037                                    return;
038                            }
039                                    try {
040                        updateComboModel(combo, (List<?>)getter.invoke(model), nullValue);
041                                    } catch (Exception ex) {
042                            Wiring.logger.error("exception in JList binding", ex); 
043                    }
044                            }
045                    };
046                    model.bind(binding);
047            return Collections.singleton(binding);
048        }
049            
050            private void updateComboModel(JComboBox combo, List<?> newContents, String nullValue) {
051                    int selectedIndex = combo.getSelectedIndex();
052                    Object selected = combo.getSelectedItem();
053                    DefaultComboBoxModel comboModel = new DefaultComboBoxModel();
054                    if (nullValue != null) {
055                            comboModel.addElement(nullValue);
056                    }
057                    for (Object obj : newContents) {
058                            comboModel.addElement(obj);
059                    }
060                    if (comboModel.getIndexOf(selected) != -1) {
061                            comboModel.setSelectedItem(selected);
062                    } else {
063                            if (comboModel.getSize() > selectedIndex) {
064                                    comboModel.setSelectedItem(combo.getItemAt(selectedIndex));
065                            } else if (comboModel.getSize() > 0) {
066                                    comboModel.setSelectedItem(combo.getItemAt(comboModel.getSize() - 1));
067                            }
068                    }
069                    combo.setModel(comboModel);
070            }
071    }
