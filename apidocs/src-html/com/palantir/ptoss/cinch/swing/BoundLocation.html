014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.awt.Component;
017    import java.awt.Point;
018    import java.awt.event.ComponentAdapter;
019    import java.awt.event.ComponentEvent;
020    import java.beans.IntrospectionException;
021    import java.lang.annotation.ElementType;
022    import java.lang.annotation.Retention;
023    import java.lang.annotation.RetentionPolicy;
024    import java.lang.annotation.Target;
025    import java.lang.reflect.Field;
026    import java.util.Collection;
027    import java.util.Collections;
028    import java.util.List;
029    
030    import org.apache.log4j.Logger;
031    
032    import com.google.common.collect.Lists;
033    import com.palantir.ptoss.cinch.core.*;
034    import com.palantir.ptoss.util.Throwables;
035    
036    /**
037     * A binding for an interface component that has a location (setLocation/getLocation methods).
038     * 
039     * @see Point
040     */
041    @Retention(RetentionPolicy.RUNTIME)
042    @Target({ElementType.FIELD})
043    public @interface BoundLocation {
044            /**
045             * The model property to bind to.
046             * @return
047             */
048            String to();
049            
050            /**
051             * When this binding should occur.
052             */
053            String on() default "";
054            
055            /**
056             * Inner utility class that performs the runtime wiring of all {@link BoundLocation} bindings.
057             * 
058             * @see Bindings#STANDARD_BINDINGS
059             */
060            public static class Wiring implements BindingWiring {
061                    private static final Logger logger = Logger.getLogger(BoundLocation.class);
062                    
063                    public Collection<Binding> wire(BindingContext context) {
064                            List<Field> boundFields = context.getAnnotatedFields(BoundLocation.class);
065                            List<Binding> bindings = Lists.newArrayList();
066                        for (Field field : boundFields) {
067                            BoundLocation bound = field.getAnnotation(BoundLocation.class);
068                            try {
069                                    bindings.addAll(wire(bound, context, field));
070                            } catch (Exception e) {
071                                    throw Throwables.throwUncheckedException(e);
072                            }
073                        }
074                        return bindings;
075                    }
076                    
077                    private static Collection<Binding> wire(BoundLocation bound, BindingContext context, Field field) throws IntrospectionException {
078                            String target = bound.to();
079                final ObjectFieldMethod setter = context.findSetter(target);
080                final ObjectFieldMethod getter = context.findGetter(target);
081                if (setter == null || getter == null) {
082                    throw new IllegalArgumentException("could not finder setter/getter for @BoundLocation: " + field); 
083                }
084                final BindableModel model1 = context.getFieldObject(setter.getField(), BindableModel.class);
085                final BindableModel model2 = context.getFieldObject(getter.getField(), BindableModel.class);
086                assert model1 == model2;
087                            if (Component.class.isAssignableFrom(field.getType())) {
088                                    final Component comp = context.getFieldObject(field, Component.class);
089                                    return bindComponent(bound, context, comp, setter, getter, model1);
090                            } else {
091                                    throw new IllegalArgumentException("don't know how to wire up @BoundLocation field: " + field.getName());       
092                            }
093                    }
094    
095                    private static Collection<Binding> bindComponent(BoundLocation bound,
096                                    BindingContext context,
097                                    final Component comp,
098                    final ObjectFieldMethod setter, final ObjectFieldMethod getter,
099                    final BindableModel model1) {
100                    
101                            String on = bound.on();
102                            final Object onObject = context.evalOnObject(on, model1);
103                            
104                            comp.addComponentListener(new ComponentAdapter() {
105                                    @Override
106                                    public void componentMoved(ComponentEvent e) {
107                                            try {
108                                            setter.getMethod().invoke(model1, comp.getLocation());
109                        } catch (Exception ex) {
110                            logger.error("could not invoke Component binding", ex); 
111                        }
112                                    }
113                            });
114                            
115                    Binding binding = new Binding() {
116                            public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
117                                    if (!BindingContext.isOn(onObject, changed)) {
118                                            return;
119                                    }
120                                    try {
121                                Point point = (Point)getter.getMethod().invoke(model1);
122                                comp.setLocation(point);
123                                    } catch (Exception ex) {
124                                    logger.error("could not invoke Component binding", ex); 
125                            }
126                        }
127                    };
128                    model1.bind(binding);
129                    return Collections.singleton(binding);
130            }
131            }
132    }
