001    package com.palantir.ptoss.cinch.swing;
002    
003    import java.beans.IntrospectionException;
004    import java.lang.reflect.Field;
005    import java.util.Collection;
006    
007    import javax.swing.JSlider;
008    import javax.swing.event.ChangeEvent;
009    import javax.swing.event.ChangeListener;
010    
011    import com.google.common.collect.ImmutableList;
012    import com.palantir.ptoss.cinch.core.*;
013    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
014    import com.palantir.ptoss.util.Mutator;
015    
016    /**
017     * A {@link WiringHarness} for binding a {@link JSlider} to an {@link Integer} value in a 
018     * {@link BindableModel}.
019     */
020    public class JSliderWiringHarness implements WiringHarness<Bound, Field> {
021            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) throws IllegalAccessException, IntrospectionException {
022                    String target = bound.to();
023                    Mutator mutator = Mutator.create(context, target);
024                    JSlider slider = context.getFieldObject(field, JSlider.class); 
025            return ImmutableList.of(bindJSlider(mutator, slider));
026        }
027            
028            public static Binding bindJSlider(final Mutator mutator, final JSlider slider) { 
029                final ChangeListener changeListener = new ChangeListener() {
030                    public void stateChanged(ChangeEvent e) {
031                            try {
032                                    if (!slider.getValueIsAdjusting()) {
033                                            mutator.set(slider.getValue());
034                                    }
035                    } catch (Exception ex) {
036                            Wiring.logger.error("exception in JSlider binding", ex); 
037                    }            
038                    }
039            };
040                    slider.addChangeListener(changeListener);
041                Binding binding = new Binding() {
042                    public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
043                                    try {
044                                            Integer val = (Integer)mutator.get();
045                                            if (val == null) {
046                                                    val = -1;
047                                            }
048                                            if (!val.equals(slider.getValue())) {
049                                                    slider.removeChangeListener(changeListener);
050                                                    slider.setValue(val);
051                                                    slider.addChangeListener(changeListener);
052                                            }
053                                    } catch (Exception ex) {
054                                            Wiring.logger.error("exception in JSlider binding", ex); 
055                                    }
056                    }
057                };
058                mutator.getModel().bind(binding);
059                return binding;
060            }
061    }
