014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.awt.event.ActionEvent;
017    import java.awt.event.ActionListener;
018    import java.beans.IntrospectionException;
019    import java.lang.reflect.Field;
020    import java.util.Collection;
021    
022    import javax.swing.AbstractButton;
023    import javax.swing.JToggleButton;
024    
025    import com.google.common.collect.ImmutableList;
026    import com.palantir.ptoss.cinch.core.*;
027    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
028    import com.palantir.ptoss.util.Mutator;
029    import com.palantir.ptoss.util.Reflections;
030    
031    /**
032     * A {@link WiringHarness} for binding a {@link JToggleButton} to an {@link Enum} value in a 
033     * {@link BindableModel}.
034     */
035    public class JToggleButtonWiringHarness implements WiringHarness<Bound, Field> {
036            public Collection<Binding> wire(Bound bound, BindingContext context, Field field)
037                    throws IllegalAccessException, IntrospectionException {
038                    String target = bound.to();
039                    Mutator mutator = Mutator.create(context, target);
040                    JToggleButton toggle = context.getFieldObject(field, JToggleButton.class);
041    
042                    Class<?>[] paramTypes = mutator.getSetter().getMethod().getParameterTypes();
043                    if (paramTypes.length == 1 && paramTypes[0].isEnum()) {
044                            Class<?> enumType = paramTypes[0];
045                            String value = bound.value();
046                            return ImmutableList.of(bindJToggleButtonToEnum(value, enumType, mutator, toggle));
047                    } else if (paramTypes.length == 1 && paramTypes[0] == boolean.class) {
048                            return ImmutableList.of(JCheckBoxWiringHarness.bindJCheckBox(mutator, toggle));
049                    } else {
050                            throw new BindingException("can only bind JToggleButtons to enums or booleans"); //$NON-NLS-1$
051                    }
052            }
053    
054            public static Binding bindJToggleButtonToEnum(final String value, final Class<?> enumType, 
055                            final Mutator mutator, final AbstractButton button) {
056                    final Object enumValue = Reflections.evalEnum(enumType, value);
057                    button.addActionListener(new ActionListener() {
058                            public void actionPerformed(ActionEvent e) {
059                                    try {
060                                            mutator.set(enumValue);
061                                    } catch (Exception ex) {
062                                            Wiring.logger.error("exception in JToggleButton binding", ex); //$NON-NLS-1$
063                                    }
064                            }
065                    });
066    
067                    Binding binding = new Binding() {
068                            public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
069                                    try {
070                                            button.setSelected(mutator.get() == enumValue);
071                                    } catch (Exception ex) {
072                                            Wiring.logger.error("exception in JToggleButton binding", ex); //$NON-NLS-1$
073                                    }
074                            }
075                    };
076                    mutator.getModel().bind(binding);
077                    return binding;
078            }
079    }
