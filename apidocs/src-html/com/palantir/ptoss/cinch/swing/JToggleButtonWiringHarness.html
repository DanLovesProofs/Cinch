001    package com.palantir.ptoss.cinch.swing;
002    
003    import java.awt.event.ActionEvent;
004    import java.awt.event.ActionListener;
005    import java.beans.IntrospectionException;
006    import java.lang.reflect.Field;
007    import java.util.Collection;
008    
009    import javax.swing.AbstractButton;
010    import javax.swing.JToggleButton;
011    
012    import com.google.common.collect.ImmutableList;
013    import com.palantir.ptoss.cinch.core.*;
014    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
015    import com.palantir.ptoss.util.Mutator;
016    import com.palantir.ptoss.util.Reflections;
017    
018    /**
019     * A {@link WiringHarness} for binding a {@link JToggleButton} to an {@link Enum} value in a 
020     * {@link BindableModel}.
021     */
022    public class JToggleButtonWiringHarness implements WiringHarness<Bound, Field> {
023            public Collection<Binding> wire(Bound bound, BindingContext context, Field field)
024                    throws IllegalAccessException, IntrospectionException {
025                    String target = bound.to();
026                    Mutator mutator = Mutator.create(context, target);
027                    JToggleButton toggle = context.getFieldObject(field, JToggleButton.class);
028    
029                    Class<?>[] paramTypes = mutator.getSetter().getMethod().getParameterTypes();
030                    if (paramTypes.length == 1 && paramTypes[0].isEnum()) {
031                            Class<?> enumType = paramTypes[0];
032                            String value = bound.value();
033                            return ImmutableList.of(bindJToggleButtonToEnum(value, enumType, mutator, toggle));
034                    } else if (paramTypes.length == 1 && paramTypes[0] == boolean.class) {
035                            return ImmutableList.of(JCheckBoxWiringHarness.bindJCheckBox(mutator, toggle));
036                    } else {
037                            throw new BindingException("can only bind JToggleButtons to enums or booleans"); //$NON-NLS-1$
038                    }
039            }
040    
041            public static Binding bindJToggleButtonToEnum(final String value, final Class<?> enumType, 
042                            final Mutator mutator, final AbstractButton button) {
043                    final Object enumValue = Reflections.evalEnum(enumType, value);
044                    button.addActionListener(new ActionListener() {
045                            public void actionPerformed(ActionEvent e) {
046                                    try {
047                                            mutator.set(enumValue);
048                                    } catch (Exception ex) {
049                                            Wiring.logger.error("exception in JToggleButton binding", ex); //$NON-NLS-1$
050                                    }
051                            }
052                    });
053    
054                    Binding binding = new Binding() {
055                            public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
056                                    try {
057                                            button.setSelected(mutator.get() == enumValue);
058                                    } catch (Exception ex) {
059                                            Wiring.logger.error("exception in JToggleButton binding", ex); //$NON-NLS-1$
060                                    }
061                            }
062                    };
063                    mutator.getModel().bind(binding);
064                    return binding;
065            }
066    }
