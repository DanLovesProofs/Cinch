014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.awt.event.FocusEvent;
017    import java.awt.event.FocusListener;
018    import java.lang.annotation.ElementType;
019    import java.lang.annotation.Retention;
020    import java.lang.annotation.RetentionPolicy;
021    import java.lang.annotation.Target;
022    import java.lang.reflect.Field;
023    import java.lang.reflect.InvocationTargetException;
024    import java.lang.reflect.Method;
025    import java.util.Collection;
026    import java.util.List;
027    
028    import org.apache.commons.lang.StringUtils;
029    import org.apache.log4j.Logger;
030    
031    import com.google.common.collect.ImmutableList;
032    import com.palantir.ptoss.cinch.core.*;
033    
034    /**
035     * A binding that will call one method when the bound component loses focus, and another method when
036     * focus is gained.
037     */
038    @Retention(RetentionPolicy.RUNTIME)
039    @Target({ElementType.FIELD})
040    public @interface OnFocusChange {
041            String lost() default ""; 
042            String gained() default ""; 
043    
044            /**
045             *      Inner utility class that performs the runtime wiring of all {@link OnFocusChange} bindings.
046             */
047            static class Wiring implements BindingWiring {
048                    private static final Logger logger = Logger.getLogger(OnFocusChange.class);
049                    
050                    private static String normalizeString(String string) {
051                            if (string == null || string.trim().length() == 0) {
052                                    return null;
053                            }
054                            return string;
055                    }
056                    
057                    public Collection<Binding> wire(BindingContext context) {
058                            List<Field> actions = context.getAnnotatedFields(OnFocusChange.class);
059                            for (Field field : actions) {
060                                    OnFocusChange focusChange = field.getAnnotation(OnFocusChange.class);
061                                    String lost = normalizeString(focusChange.lost());
062                                    String gained = normalizeString(focusChange.gained());
063                                    if (StringUtils.isEmpty(lost) && StringUtils.isEmpty(gained)) {
064                                            throw new BindingException("either lost or gained must be specified on @OnFocusChange on " + field.getName()); 
065                                    }
066                                    try {
067                                            wire(lost, gained, field, context);
068                                    } catch (Exception e) {
069                                            throw new BindingException("could not wire up @OnFocusChange " + field.getName(), e); 
070                                    }
071                            }
072                            return ImmutableList.of();
073                    }
074                    
075                    private static void wire(String lost, String gained, Field field, BindingContext context) 
076                            throws SecurityException, NoSuchMethodException, IllegalArgumentException, IllegalAccessException, InvocationTargetException {
077                Method aflMethod = field.getType().getMethod("addFocusListener", FocusListener.class); 
078                            if (aflMethod != null) {
079                            Object actionObject = context.getFieldObject(field, Object.class);
080                            final ObjectFieldMethod lostOFM;
081                            if (lost == null) {
082                                    lostOFM = null;
083                            } else {
084                                    lostOFM = context.getBindableMethod(lost);
085                                    if (lostOFM == null) {
086                                            throw new BindingException("could not find bindable method: " + lost); 
087                                    }       
088                            }
089                            final ObjectFieldMethod gainedOFM;
090                            if (gained == null) {
091                                    gainedOFM = null;
092                            } else {
093                                    gainedOFM = context.getBindableMethod(gained);
094                                    if (gainedOFM == null) {
095                                            throw new BindingException("could not find bindable method: " + gained); 
096                                    }       
097                            }
098                            FocusListener focusListener = new FocusListener() {
099                                            public void focusGained(FocusEvent e) {
100                                                    try {
101                                                            if (gainedOFM != null) {
102                                                                    boolean accessible = gainedOFM.getMethod().isAccessible();
103                                                                    gainedOFM.getMethod().setAccessible(true);
104                                                                    gainedOFM.getMethod().invoke(gainedOFM.getObject());
105                                                                    gainedOFM.getMethod().setAccessible(accessible);
106                                                            }
107                            } catch (Exception ex) {
108                                    logger.error("exception during focusGained firing", ex); 
109                            }
110                        }
111    
112                                            public void focusLost(FocusEvent e) {
113                                                    try {
114                                                            if (lostOFM != null) {
115                                                                    boolean accessible = lostOFM.getMethod().isAccessible();
116                                                                    lostOFM.getMethod().setAccessible(true);
117                                                                    lostOFM.getMethod().invoke(lostOFM.getObject());
118                                                                    lostOFM.getMethod().setAccessible(accessible);
119                                                            }
120                            } catch (Exception ex) {
121                                    logger.error("exception during focusLost firing", ex); 
122                            }                                   
123                        }
124                    };
125                    aflMethod.invoke(actionObject, focusListener);
126                    }
127            }
128            }
129    }
