014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.beans.IntrospectionException;
017    import java.lang.annotation.ElementType;
018    import java.lang.annotation.Retention;
019    import java.lang.annotation.RetentionPolicy;
020    import java.lang.annotation.Target;
021    import java.lang.reflect.Field;
022    import java.lang.reflect.Method;
023    import java.util.Collection;
024    import java.util.Collections;
025    import java.util.List;
026    
027    import org.apache.log4j.Logger;
028    
029    import com.google.common.collect.Lists;
030    import com.palantir.ptoss.cinch.core.BindableModel;
031    import com.palantir.ptoss.cinch.core.Binding;
032    import com.palantir.ptoss.cinch.core.BindingContext;
033    import com.palantir.ptoss.cinch.core.BindingException;
034    import com.palantir.ptoss.cinch.core.BindingWiring;
035    import com.palantir.ptoss.cinch.core.ModelUpdate;
036    import com.palantir.ptoss.cinch.core.ObjectFieldMethod;
037    
038    /**
039     * A binding that will set the Visible state of the annotated component to the state of a model
040     * boolean. The component must have a "setVisible" method that takes a boolean.
041     */
042    @Retention(RetentionPolicy.RUNTIME)
043    @Target({ElementType.FIELD})
044    public @interface VisibleIf {
045            /**
046             * Enum to specify if normal or inverted comparisons should be used.
047             */
048            public enum Type { NORMAL, INVERTED };
049            /**
050             * The model boolean property to bind to.
051             */
052            String to();
053    
054            /**
055             * Whether or not to invert the boolean.
056             */
057            Type type() default Type.NORMAL;
058    
059            /**
060             * Inner utility class that performs the runtime wiring of all {@link VisibleIf} bindings. 
061             */
062            public static class Wiring implements BindingWiring {
063                    private static final Logger logger = Logger.getLogger(VisibleIf.class);
064    
065                    public Collection<Binding> wire(final BindingContext context) {
066                            final List<Field> actions = context.getAnnotatedFields(VisibleIf.class);
067                            final List<Binding> bindings = Lists.newArrayList();
068                            for (final Field field : actions) {
069                                    final VisibleIf action = field.getAnnotation(VisibleIf.class);
070                                    final String to = action.to();
071                                    final boolean invert = (action.type() == Type.INVERTED);
072                                    try {
073                                            bindings.addAll(wire(to, field, context, invert));
074                                    } catch (final Exception e) {
075                                            throw new BindingException("could not wire up @VisibleIf on " + field.getName(), e); 
076                                    }
077                            }
078                            return bindings;
079                    }
080    
081                    private static Collection<Binding> wire(final String to, final Field field, final BindingContext context, final boolean invert)
082                            throws SecurityException, NoSuchMethodException, IllegalArgumentException, IntrospectionException {
083                            final Method setVisibleMethod = field.getType().getMethod("setVisible", boolean.class); 
084                            if (setVisibleMethod == null) {
085                                    throw new BindingException("no setVisible call on VisibleIf field: " + field); 
086                            }
087                            final Object setVisibleObject = context.getFieldObject(field, Object.class);
088                            final ObjectFieldMethod getter = context.findGetter(to);
089                            if (getter == null) {
090                                    throw new BindingException("could not find bindable property: " + to); 
091                            }
092                            if (getter.getMethod().getReturnType() != boolean.class) {
093                                    throw new BindingException("VisibleIf binding must return boolean: " + to); 
094                            }
095                            final Binding binding = new Binding() {
096                                    public <T extends Enum<?> & ModelUpdate> void update(final T... changed) {
097                                            try {
098                                                    getter.getMethod().setAccessible(true);
099                                                    boolean visible = (Boolean)getter.getMethod().invoke(getter.getObject());
100                                                    if (invert) {
101                                                            visible = !visible;
102                                                    }
103                                                    setVisibleMethod.invoke(setVisibleObject, visible);
104                                            } catch (final Exception e) {
105                                                    Wiring.logger.error("exception during VisibleIf binding", e); 
106                                            }
107                                    }
108                            };
109                            ((BindableModel)getter.getObject()).bind(binding);
110                            return Collections.singleton(binding);
111                    }
112            }
113    }
