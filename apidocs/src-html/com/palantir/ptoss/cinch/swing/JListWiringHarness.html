001    package com.palantir.ptoss.cinch.swing;

003    import java.beans.IntrospectionException;
004    import java.lang.reflect.Field;
005    import java.util.Collection;
006    import java.util.List;
007    import java.util.Set;

009    import javax.swing.DefaultListModel;
010    import javax.swing.JList;
011    import javax.swing.ListModel;
012    import javax.swing.table.DefaultTableModel;
013    
014    import org.apache.commons.lang.ArrayUtils;
015    
016    import com.google.common.collect.ImmutableList;
017    import com.google.common.collect.Lists;
018    import com.google.common.collect.Sets;
019    import com.palantir.ptoss.cinch.core.BindableModel;
020    import com.palantir.ptoss.cinch.core.Binding;
021    import com.palantir.ptoss.cinch.core.BindingContext;
022    import com.palantir.ptoss.cinch.core.ModelUpdate;
023    import com.palantir.ptoss.cinch.core.WiringHarness;
024    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
025    import com.palantir.ptoss.util.Mutator;
026    
027    /**
028     * <p>
029     * Wires up a {@link JList} to a {@link List} in a {@link BindableModel}. Whenever the contents of
030     * the bound List change (and the BindableModel's update is called) the JList's model will be
031     * swapped out with a new model with the new contents. Selection of values is maintained between
032     * model swaps.
033     * <p>
034     * This bindings is only one way. The JList will modify itself to reflect the contents of the List
035     * in the BindableModel. However, changes to the JList's model after that will not be reflected
036     * in the model. The idea here is that if the user takes some action to remove an item from the 
037     * list then that should be accomplished by removing an item from the underlying BindableModel,
038     * not the JList.
039     * <p>
040     * Implementation notes: a {@link DefaultTableModel} is used to back the JList. Maintaining
041     * selection between model swaps is simply done with {@link Object#equals(Object)} comparisons.
042     */
043    public class JListWiringHarness implements WiringHarness<Bound, Field> {
044            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) 
045                            throws IllegalAccessException, IntrospectionException {
046                    JList list = context.getFieldObject(field, JList.class);
047                    Mutator mutator = Mutator.create(context, bound.to());
048            return ImmutableList.of(bindJList(bound, mutator, list));
049            }
050    
051            private Binding bindJList(final Bound bound, final Mutator mutator, final JList list) {
052                    final List<Object> ons = BindingContext.getOnObjects(bound.on(), mutator.getModel());
053                    Binding binding = new Binding() {
054                    public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
055                            if (!BindingContext.isOn(ons, changed)) {
056                                    return;
057                            }
058                                    try {
059                        updateListModel(list, (List<?>)mutator.get());
060                                    } catch (Exception ex) {
061                            Wiring.logger.error("exception in JList binding", ex); 
062                    }
063                            }
064                    };
065                    mutator.getModel().bind(binding);
066            return binding;
067        }
068            
069            private static void updateListModel(JList list, List<?> newContents) {
070                    if (newContents == null) {
071                            newContents = ImmutableList.of();
072                    }
073                    ListModel oldModel = list.getModel();
074                    List<Object> old = Lists.newArrayListWithExpectedSize(oldModel.getSize());
075                    for (int i = 0; i < oldModel.getSize(); i++) {
076                            old.add(oldModel.getElementAt(i));
077                    }
078                    if (old.equals(newContents)) {
079                            return;
080                    }
081                    Object[] selected = list.getSelectedValues();
082                    DefaultListModel listModel = new DefaultListModel();
083                    for (Object obj : newContents) {
084                            listModel.addElement(obj);
085                    }
086                    list.setModel(listModel);
087                    List<Integer> newIndices = Lists.newArrayListWithCapacity(selected.length);
088                    Set<Object> selectedSet = Sets.newHashSet(selected);
089                    for (int i = 0; i < listModel.size(); i++) {
090                            if (selectedSet.contains(listModel.elementAt(i))) {
091                                    newIndices.add(i);
092                            }
093                    }
094                    list.setSelectedIndices(ArrayUtils.toPrimitive(newIndices.toArray(new Integer[0])));
095            }
096    }
