001    package com.palantir.ptoss.cinch.swing;
002    
003    import java.awt.event.ActionEvent;
004    import java.awt.event.ActionListener;
005    import java.beans.IntrospectionException;
006    import java.lang.reflect.Field;
007    import java.util.Collection;
008    
009    import javax.swing.AbstractButton;
010    import javax.swing.JCheckBox;
011    
012    import com.google.common.collect.ImmutableList;
013    import com.palantir.ptoss.cinch.core.BindableModel;
014    import com.palantir.ptoss.cinch.core.Binding;
015    import com.palantir.ptoss.cinch.core.BindingContext;
016    import com.palantir.ptoss.cinch.core.ModelUpdate;
017    import com.palantir.ptoss.cinch.core.WiringHarness;
018    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
019    import com.palantir.ptoss.util.Mutator;
020    
021    /**
022     * A {@link WiringHarness} for binding a {@link JCheckBox} to a boolean value in a 
023     * {@link BindableModel}.
024     */
025    public class JCheckBoxWiringHarness implements WiringHarness<Bound, Field> {
026            
027            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) 
028                            throws IllegalAccessException, IntrospectionException {
029                    Mutator mutator = Mutator.create(context, bound.to());
030                    JCheckBox checkbox = context.getFieldObject(field, JCheckBox.class);
031            return ImmutableList.of(bindJCheckBox(mutator, checkbox));
032        }
033            
034            public static Binding bindJCheckBox(
035                            final Mutator mutator, final AbstractButton checkbox) {
036                checkbox.addActionListener(new ActionListener() {
037                    public void actionPerformed(ActionEvent e) {
038                            try {
039                                    mutator.set(checkbox.isSelected());
040                            } catch (Exception ex) {
041                            Wiring.logger.error("exception in JCheckBox binding", ex); 
042                    }
043                    }
044                });
045                Binding binding = new Binding() {
046                    public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
047                                    try {
048                                            checkbox.setSelected((Boolean)mutator.get());
049                    } catch (Exception ex) {
050                            Wiring.logger.error("exception in JCheckBox binding", ex); 
051                    }
052                }
053                };
054                    mutator.getModel().bind(binding);
055                    return binding;
056        }
057    }
