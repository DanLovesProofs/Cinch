016    import java.awt.event.ItemEvent;
017    import java.awt.event.ItemListener;

029    import javax.swing.JComboBox;
030    import javax.swing.JList;
031    import javax.swing.event.ListSelectionEvent;
032    import javax.swing.event.ListSelectionListener;
033    
034    import org.apache.log4j.Logger;

036    import com.google.common.base.Objects;
037    import com.google.common.collect.ImmutableList;
038    import com.google.common.collect.Iterables;
039    import com.google.common.collect.Lists;
040    import com.google.common.collect.Sets;

048    /**
049     * A binding for an interface component that has a selection like a {@link JList} or
050     * {@link JComboBox}.
051     */
052    @Retention(RetentionPolicy.RUNTIME)
053    @Target({ElementType.FIELD})
054    public @interface BoundSelection {
055            /**
056             * The model property to bind to.
057             */
058            String to();
059            
060            /**
061             * When this binding should occur.
062             */
063            String[] on() default "";
064            
065            /**
066             * How to render the null value in a JComboBox.
067             */
068            String nullValue() default ""; 
069            
070            /**
071             * Whether or not a JList should bind with multiselect.
072             */
073            boolean multi() default false;
074            
075            /**
076             * Inner utility class that performs the runtime wiring of all {@link BoundSelection} bindings.
077             */
078            public static class Wiring implements BindingWiring {
079                    private static final Logger logger = Logger.getLogger(BoundSelection.class);
080                    
081                    public Collection<Binding> wire(BindingContext context) {
082                            List<Field> boundFields = context.getAnnotatedFields(BoundSelection.class);
083                            List<Binding> bindings = Lists.newArrayList();
084                        for (Field field : boundFields) {
085                            BoundSelection bound = field.getAnnotation(BoundSelection.class);
086                            try {
087                                    bindings.addAll(wire(bound, context, field));
088                            } catch (Exception e) {
089                                    throw Throwables.throwUncheckedException(e);
090                            }
091                        }
092                        return bindings;
093                    }
094                    
095                    private static Collection<Binding> wire(BoundSelection bound, BindingContext context, Field field) throws IntrospectionException {
096                            String target = bound.to();
097                            Mutator mutator = Mutator.create(context, target);
098                            if (JList.class.isAssignableFrom(field.getType())) {
099                                    final JList list = context.getFieldObject(field, JList.class);
100                                    return bindJList(bound, mutator, list);
101                            } else if (JComboBox.class.isAssignableFrom(field.getType())) {
102                                    final JComboBox combo = context.getFieldObject(field, JComboBox.class);
103                                    final String nullValue = (String)Bound.Utilities.getNullValue(context, bound.nullValue());
104                                    return bindJComboBox(bound, mutator, combo, nullValue);
105                            } else {
106                                    throw new IllegalArgumentException("don't know how to wire up @BoundSelection field: " + field.getName());      
107                            }
108                    }
109    
110                    private static Collection<Binding> bindJList(BoundSelection bound,
111                                    final Mutator mutator,
112                                    final JList list) {
113                    final boolean multi = bound.multi();
114                            final List<Object> ons = BindingContext.getOnObjects(bound.on(), mutator.getModel());
115                            
116                    list.addListSelectionListener(new ListSelectionListener() {
117                            public void valueChanged(ListSelectionEvent e) {
118                                    if (!e.getValueIsAdjusting()) {
119                                            try {
120                                                    if (multi) {
121                                                            mutator.set(ImmutableList.copyOf(list.getSelectedValues()));
122                                                    } else {
123                                                            mutator.set(list.getSelectedValue());
124                                                    }
125                                } catch (Exception ex) {
126                                    logger.error("could not invoke JList binding", ex); 
127                                }
128                                    }
129                        }
130                    });
131                    
132                    Binding binding = new Binding() {
133                            public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
134                                    if (!BindingContext.isOn(ons, changed)) {
135                                            return;
136                                    }
137                                    try {
138                                            if (multi) {
139                                                    Object[] selVals = list.getSelectedValues();
140                                                    ImmutableList<Object> listCurrent = ImmutableList.copyOf(selVals);
141                                                    Collection<?> current = (Collection<?>)mutator.get();
142                                                    if (current == null) {
143                                                            current = ImmutableList.of();
144                                                    }
145                                                    if (!Iterables.elementsEqual(listCurrent, current)) {
146                                                            Set<?> currentSet = Sets.newHashSet(current);
147                                                            List<Integer> selIndices = Lists.newArrayList();
148                                                            for (int i = 0; i < list.getModel().getSize(); i++) {
149                                                                    if (currentSet.contains(list.getModel().getElementAt(i))) {
150                                                                            selIndices.add(i);
151                                                                    }
152                                                            }
153                                                            int[] sel = new int[selIndices.size()];
154                                                            int i = 0;
155                                                            for (Integer index : selIndices) {
156                                                                    sel[i++] = index;
157                                                            }
158                                                            list.setSelectedIndices(sel);
159                                                    }
160                                            } else {
161                                Object current = mutator.get();
162                                Object listCurrent = list.getSelectedValue();
163                                if (!Objects.equal(current, listCurrent)) {
164                                    if (current == null) {
165                                            list.clearSelection();
166                                    } else {
167                                            list.setSelectedValue(current, true);
168                                    }
169                                }
170                                            }
171                                    } catch (Exception ex) {
172                                    logger.error("could not invoke JList binding", ex); 
173                            }
174                        }
175                    };
176                    mutator.getModel().bind(binding);
177                    return Collections.singleton(binding);
178            }
179                    
180                    private static Collection<Binding> bindJComboBox(final BoundSelection bound,
181                                    final Mutator mutator,
182                                    final JComboBox combo,
183                    final String nullValue) {
184                    
185                            combo.addItemListener(new ItemListener() {
186                                    public void itemStateChanged(ItemEvent e) {
187                                            try {
188                                                    Object current = mutator.get();
189                                                    Object newValue = combo.getSelectedItem();
190                                                    if (newValue.equals(nullValue)) {
191                                                            newValue = null;
192                                                    }
193                                                    if (!Objects.equal(current, newValue)) {
194                                                            mutator.set(newValue);
195                                                    }
196                                            } catch (Exception ex) {
197                                                    logger.error("could not invoke JComboBox binding", ex); 
198                                            }
199                                    }
200                            });
201                    
202                    Binding binding = new Binding() {
203                            public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
204                                    try {
205                                Object current = mutator.get();
206                                if (current == null) {
207                                    current = nullValue;
208                                }
209                                if (!Objects.equal(combo.getSelectedItem(), current)) {
210                                    combo.setSelectedItem(current);
211                                }
212                                    } catch (Exception ex) {
213                                    logger.error("could not invoke JComboBox binding", ex); 
214                            }
215                        }
216                    };
217                    mutator.getModel().bind(binding);
218                    return Collections.singleton(binding);
219            }
220            }
