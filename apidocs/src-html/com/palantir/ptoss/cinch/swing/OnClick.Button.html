014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.awt.event.MouseAdapter;
017    import java.awt.event.MouseEvent;
018    import java.awt.event.MouseListener;
019    import java.lang.annotation.ElementType;
020    import java.lang.annotation.Retention;
021    import java.lang.annotation.RetentionPolicy;
022    import java.lang.annotation.Target;
023    import java.lang.reflect.Field;
024    import java.lang.reflect.InvocationTargetException;
025    import java.lang.reflect.Method;
026    import java.util.Collection;
027    import java.util.List;
028    
029    import org.apache.log4j.Logger;
030    
031    import com.google.common.collect.ImmutableList;
032    import com.palantir.ptoss.cinch.core.Binding;
033    import com.palantir.ptoss.cinch.core.BindingContext;
034    import com.palantir.ptoss.cinch.core.BindingException;
035    import com.palantir.ptoss.cinch.core.BindingWiring;
036    import com.palantir.ptoss.cinch.core.ObjectFieldMethod;
037    
038    /**
039     * A binding that will call a method when the on-click action occurs on this component.
040     */
041    @Retention(RetentionPolicy.RUNTIME)
042    @Target({ElementType.FIELD})
043    public @interface OnClick {
044            String call();
045            int count() default 1;
046            Button button() default Button.LEFT;
047            /**
048             * Enum specifying which mouse button is being bound by this binding.
049             */     
050            public enum Button {
051                    LEFT(MouseEvent.BUTTON1), CENTER(MouseEvent.BUTTON2), RIGHT(MouseEvent.BUTTON3);
052                    
053                    private final int constant;
054                    
055                    private Button(int constant) {
056                            this.constant = constant;
057                    }
058                    
059                    private int getConstant() {
060                            return constant;
061                    }
062            }
063            
064            /**
065             *      Inner utility class that performs the runtime wiring of all {@link OnClick} bindings.
066             */
067            static class Wiring implements BindingWiring {
068                    private static final Logger logger = Logger.getLogger(OnClick.class);
069                    
070                    public Collection<Binding> wire(BindingContext context) {
071                            List<Field> actions = context.getAnnotatedFields(OnClick.class);
072                            for (Field field : actions) {
073                                    OnClick onClick = field.getAnnotation(OnClick.class);
074                                    try {
075                                            wire(onClick, field, context);
076                                    } catch (Exception e) {
077                                            throw new BindingException("could not wire up @OnClick on " + field.getName(), e); 
078                                    }               
079                            }
080                            return ImmutableList.of();
081                    }
082                    
083                    private static void wire(final OnClick onClick, Field field, BindingContext context) 
084                            throws SecurityException, NoSuchMethodException, IllegalArgumentException, IllegalAccessException, InvocationTargetException {
085                            String call = onClick.call();
086                            if (call == null) {
087                                    throw new BindingException("call on @OnClick on " + field.getName() + " must be specified"); 
088                            }
089                Method amlMethod = field.getType().getMethod("addMouseListener", MouseListener.class); 
090                            if (amlMethod != null) {
091                            Object actionObject = context.getFieldObject(field, Object.class);
092                            final ObjectFieldMethod ofm = context.getBindableMethod(call);
093                            if (ofm == null) {
094                                    throw new BindingException("could not find bindable method: " + call); 
095                            }
096                            MouseListener mouseListener = new MouseAdapter() {
097                            @Override
098                        public void mouseClicked(MouseEvent e) {
099                                    if (e.getButton() != onClick.button().getConstant() || e.getClickCount() != onClick.count()) {
100                                            return;
101                                    }
102                                    try {
103                                            ofm.getMethod().setAccessible(true);
104                                ofm.getMethod().invoke(ofm.getObject());
105                            } catch (Exception ex) {
106                                    logger.error("exception during action firing", ex); 
107                            }
108                            }
109                    };
110                    amlMethod.invoke(actionObject, mouseListener);
111                    }
112            }
113            }
114    }
