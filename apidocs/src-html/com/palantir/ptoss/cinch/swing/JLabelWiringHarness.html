014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.beans.IntrospectionException;
017    import java.lang.reflect.Field;
018    import java.util.Collection;
019    
020    import javax.swing.JLabel;
021    
022    import com.google.common.collect.ImmutableList;
023    import com.palantir.ptoss.cinch.core.*;
024    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
025    import com.palantir.ptoss.util.Mutator;
026    
027    /**
028     * A {@link WiringHarness} for binding a {@link JLabel} to a value in a {@link BindableModel}.
029     */
030    public class JLabelWiringHarness implements WiringHarness<Bound, Field> {
031            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) 
032                            throws IllegalAccessException, IntrospectionException {
033                    JLabel label = context.getFieldObject(field, JLabel.class);
034                    Mutator mutator = Mutator.create(context, bound.to());
035                    return ImmutableList.of(bindJLabel(mutator, label));
036            }
037            
038            public static Binding bindJLabel(final Mutator mutator, final JLabel label) {
039                    Binding binding = new Binding() {
040                            public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
041                                    try {
042                                            String text = ""; 
043                                            Object obj = mutator.get();
044                                            if (obj != null) {
045                                                    text = obj.toString();
046                                            }
047                                            label.setText(text);
048                                    } catch (Exception ex) {
049                                            Wiring.logger.error("exception in JLabel binding", ex); 
050                                    }
051                            }
052                    };
053                    mutator.getModel().bind(binding);
054                    return binding;
055            }
056    }
