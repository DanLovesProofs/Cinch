014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.lang.annotation.ElementType;
017    import java.lang.annotation.Retention;
018    import java.lang.annotation.RetentionPolicy;
019    import java.lang.annotation.Target;
020    import java.lang.reflect.Field;
021    import java.lang.reflect.InvocationTargetException;
022    import java.lang.reflect.Method;
023    import java.util.Collection;
024    import java.util.List;
025    
026    import javax.swing.JButton;
027    import javax.swing.JTextField;
028    import javax.swing.event.ChangeEvent;
029    import javax.swing.event.ChangeListener;
030    
031    import org.apache.log4j.Logger;
032    
033    import com.google.common.collect.ImmutableList;
034    import com.palantir.ptoss.cinch.core.Binding;
035    import com.palantir.ptoss.cinch.core.BindingContext;
036    import com.palantir.ptoss.cinch.core.BindingException;
037    import com.palantir.ptoss.cinch.core.BindingWiring;
038    import com.palantir.ptoss.cinch.core.ObjectFieldMethod;
039    
040    /**
041     * A component binding that will call a method when an action occurs.  This can be applied to 
042     * any object that has an "addChangeListener" method that takes an {@link ChangeListener}.
043     * Normally used for any sort of {@link JButton} or for a {@link JTextField}.
044     */
045    @Retention(RetentionPolicy.RUNTIME)
046    @Target({ElementType.FIELD})
047    public @interface OnChange {
048            /**
049             * The name of the method to call when the action occurs. Must be accessible in the 
050             * {@link BindingContext}.
051             */
052            String call();
053            
054            /**
055             * A boolean indicating whether or not we want to call this when the values
056             * is adjusting
057             */
058            boolean onAdjust() default false;
059    
060            /**
061             *      Inner utility class that performs the runtime wiring of all {@link OnChange} bindings.
062             */
063            static class Wiring implements BindingWiring {
064                    private static final Logger logger = Logger.getLogger(OnChange.class);
065                    
066                    public Collection<Binding> wire(BindingContext context) {
067                            List<Field> actions = context.getAnnotatedFields(OnChange.class);
068                            for (Field field : actions) {
069                                    OnChange change = field.getAnnotation(OnChange.class);
070                                    try {
071                                            wire(change, field, context);
072                                    } catch (Exception e) {
073                                            throw new BindingException("could not wire up @OnChange on " + field.getName(), e); 
074                                    }
075                            }
076                            return ImmutableList.of();
077                    }
078                    
079                    private static void wire(final OnChange change, Field field, BindingContext context) 
080                            throws SecurityException, NoSuchMethodException, IllegalArgumentException, IllegalAccessException, InvocationTargetException {
081    
082                            String call = change.call();
083                            if (call == null) {
084                                    throw new BindingException("call on @OnChange on " + field.getName() + " must be specified");
085                            }
086                            Method _adjustMethod = null;
087                            try {
088                                    _adjustMethod = field.getType().getMethod("getValueIsAdjusting"); 
089                            } catch(NoSuchMethodException nsme) {
090                                    // not a problem, leave value as null;
091                            }
092                            
093                final Method adjustMethod = _adjustMethod; 
094                final Method addChangeMethod = field.getType().getMethod("addChangeListener", ChangeListener.class); 
095                            if (addChangeMethod != null) {
096                                    final Object changeObject = context.getFieldObject(field, Object.class);
097                            final ObjectFieldMethod ofm = context.getBindableMethod(call);
098                            if (ofm == null) {
099                                    throw new BindingException("could not find bindable method: " + call); 
100                            }
101                            final ChangeListener changeListener = new ChangeListener() {
102                                            public void stateChanged(ChangeEvent e) {
103                                                    try {
104                                                            if (adjustMethod != null && !change.onAdjust()) {
105                                                                    if ((Boolean) adjustMethod.invoke(changeObject)) return;
106                                                            }
107                                                            ofm.getMethod().invoke(ofm.getObject());
108                                                    } catch (InvocationTargetException itex) {
109                                                            logger.error("exception during action firing", itex.getCause()); 
110                                                    } catch (Exception ex) {
111                                                            logger.error("exception during action firing", ex); 
112                                                    }
113                                            }
114                                    };
115                    addChangeMethod.invoke(changeObject, changeListener);
116                    }
117            }
118            }
119    }
