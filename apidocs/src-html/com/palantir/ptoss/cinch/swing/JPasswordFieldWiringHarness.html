014    package com.palantir.ptoss.cinch.swing;
015    
016    import java.beans.IntrospectionException;
017    import java.lang.reflect.Field;
018    import java.lang.reflect.Method;
019    import java.util.Arrays;
020    import java.util.Collection;
021    import java.util.Collections;
022    
023    import javax.swing.JPasswordField;
024    import javax.swing.event.DocumentEvent;
025    import javax.swing.event.DocumentListener;
026    
027    import org.apache.commons.lang.Validate;
028    
029    import com.palantir.ptoss.cinch.core.*;
030    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
031    
032    // The interface uses char[] but has to make a String under the covers in order to set the password.
033    // TODO (dcervelli): find a (hacky) way to set the text without throwing it into a String
034     /**
035     * A {@link WiringHarness} for binding a {@link JPasswordField} to a value in a 
036     * {@link BindableModel}.
037     */
038    public class JPasswordFieldWiringHarness implements WiringHarness<Bound, Field> {
039            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) throws IllegalAccessException, IntrospectionException {
040                    String target = bound.to();
041                    JPasswordField pwdField = context.getFieldObject(field, JPasswordField.class); 
042            ObjectFieldMethod setter = context.findSetter(target);
043            ObjectFieldMethod getter = context.findGetter(target);
044            if (setter == null || getter == null) {
045                    throw new IllegalArgumentException("could not find getter/setter for " + target); 
046            }
047            BindableModel model1 = context.getFieldObject(setter.getField(), BindableModel.class);
048            BindableModel model2 = context.getFieldObject(getter.getField(), BindableModel.class);
049            Validate.isTrue(model1 == model2);
050            // verify type parameters
051            return bindJPasswordField(model1, pwdField, getter.getMethod(), setter.getMethod());
052        }
053            
054            public static Collection<Binding> bindJPasswordField(final BindableModel model, final JPasswordField pwdField, 
055                            final Method getter, final Method setter) {
056                pwdField.getDocument().addDocumentListener(new DocumentListener() {
057                            public void removeUpdate(DocumentEvent e) {
058                                    updateModel();
059                            }
060                            
061                            public void insertUpdate(DocumentEvent e) {
062                                    updateModel();
063                            }
064                            
065                            public void changedUpdate(DocumentEvent e) {
066                                    updateModel();
067                            }
068    
069                            private void updateModel() {
070                                    try {
071                            setter.invoke(model, pwdField.getPassword());
072                            } catch (Exception ex) {
073                            Wiring.logger.error("exception in JPasswordField binding", ex); 
074                    }                       
075                }
076                    });
077                Binding binding = new Binding() {
078                    public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
079                                    try {
080                            char[] charArray = (char[])getter.invoke(model);
081                            if (charArray == null) {
082                                    charArray = new char[0]; 
083                            }
084                            if (!Arrays.equals(charArray, pwdField.getPassword())) {
085                                    pwdField.setText(String.valueOf(charArray));
086                            }
087                    } catch (Exception ex) {
088                            Wiring.logger.error("exception in JPasswordField binding", ex); 
089                    }
090                }
091                };
092                    model.bind(binding);
093                    return Collections.singleton(binding);
094        }
095    }
