001    package com.palantir.ptoss.cinch.swing;
002    
003    import java.beans.IntrospectionException;
004    import java.lang.reflect.Field;
005    import java.lang.reflect.Method;
006    import java.util.Arrays;
007    import java.util.Collection;
008    import java.util.Collections;
009    
010    import javax.swing.JPasswordField;
011    import javax.swing.event.DocumentEvent;
012    import javax.swing.event.DocumentListener;
013    
014    import org.apache.commons.lang.Validate;
015    
016    import com.palantir.ptoss.cinch.core.*;
017    import com.palantir.ptoss.cinch.swing.Bound.Wiring;
018    
019    // The interface uses char[] but has to make a String under the covers in order to set the password.
020    // TODO (dcervelli): find a (hacky) way to set the text without throwing it into a String
021     /**
022     * A {@link WiringHarness} for binding a {@link JPasswordField} to a value in a 
023     * {@link BindableModel}.
024     */
025    public class JPasswordFieldWiringHarness implements WiringHarness<Bound, Field> {
026            public Collection<Binding> wire(Bound bound, BindingContext context, Field field) throws IllegalAccessException, IntrospectionException {
027                    String target = bound.to();
028                    JPasswordField pwdField = context.getFieldObject(field, JPasswordField.class); 
029            ObjectFieldMethod setter = context.findSetter(target);
030            ObjectFieldMethod getter = context.findGetter(target);
031            if (setter == null || getter == null) {
032                    throw new IllegalArgumentException("could not find getter/setter for " + target); 
033            }
034            BindableModel model1 = context.getFieldObject(setter.getField(), BindableModel.class);
035            BindableModel model2 = context.getFieldObject(getter.getField(), BindableModel.class);
036            Validate.isTrue(model1 == model2);
037            // verify type parameters
038            return bindJPasswordField(model1, pwdField, getter.getMethod(), setter.getMethod());
039        }
040            
041            public static Collection<Binding> bindJPasswordField(final BindableModel model, final JPasswordField pwdField, 
042                            final Method getter, final Method setter) {
043                pwdField.getDocument().addDocumentListener(new DocumentListener() {
044                            public void removeUpdate(DocumentEvent e) {
045                                    updateModel();
046                            }
047                            
048                            public void insertUpdate(DocumentEvent e) {
049                                    updateModel();
050                            }
051                            
052                            public void changedUpdate(DocumentEvent e) {
053                                    updateModel();
054                            }
055    
056                            private void updateModel() {
057                                    try {
058                            setter.invoke(model, pwdField.getPassword());
059                            } catch (Exception ex) {
060                            Wiring.logger.error("exception in JPasswordField binding", ex); 
061                    }                       
062                }
063                    });
064                Binding binding = new Binding() {
065                    public <T extends Enum<?> & ModelUpdate> void update(T... changed) {
066                                    try {
067                            char[] charArray = (char[])getter.invoke(model);
068                            if (charArray == null) {
069                                    charArray = new char[0]; 
070                            }
071                            if (!Arrays.equals(charArray, pwdField.getPassword())) {
072                                    pwdField.setText(String.valueOf(charArray));
073                            }
074                    } catch (Exception ex) {
075                            Wiring.logger.error("exception in JPasswordField binding", ex); 
076                    }
077                }
078                };
079                    model.bind(binding);
080                    return Collections.singleton(binding);
081        }
082    }
