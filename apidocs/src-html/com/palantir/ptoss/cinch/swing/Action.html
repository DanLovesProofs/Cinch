043    /**
044     * <p>
045     * A component binding that will call a method when an action occurs.  This can be applied to 
046     * any object that has an "addActionListener" method that takes an {@link ActionListener}.
047     * Normally used for any sort of {@link JButton} or for a {@link JTextField}.
048     * </p>
049     * Here's an example of using @Action annotations:
050     * <pre>
051     * <code>
052     * public class BoundJCheckBoxExample {
053     *
054     *     public static class Model extends DefaultBindableModel {
055     *         private boolean state;
056     *      
057     *         public void setState(boolean state) {
058     *             this.state = state;
059     *             update();
060     *         }
061     *      
062     *         public boolean isState() {
063     *             return state;
064     *         } 
065    *      }
066     * 
067     *     public static class Controller {
068     *         private final Model model;
069     *
070     *         public Controller(Model model) {
071     *             this.model = model;
072     *         }
073     *     
074     *         public void setToTrue() {
075     *             model.setState(true);
076     *         }
077     *     
078     *         public void setToFalse() {
079     *             model.setState(false);
080     *         }
081     *     }
082     * 
083     *     private final Model model = new Model();
084     *     &#064;Bindable
085     *     private final Controller controller = new Controller(model);
086     * 
087     *     &#064;Bound(to = "state")
088     *     private final JCheckBox box = new JCheckBox("State");
089     *     &#064;Bound(to = "state")
090     *     private final JLabel stateLabel = new JLabel("?");
091     * 
092     *     &#064;Action(call = "setToTrue")
093     *     private final JButton trueButton = new JButton("Set True");
094     *     &#064;Action(call = "setToFalse")
095     *     private final JButton falseButton = new JButton("Set False");
096     * 
097     *     private final JPanel panel = new JPanel();
098     * 
099     *     private final Bindings bindings = Bindings.standard();
100     * 
101     *     public BoundJCheckBoxExample() {
102     *            // ... layout panel ...
103     *            bindings.bind(this);
104     *     }
105     * 
106     *     // main()
107     * }
108     * </pre></code>
109     * 
110     * <h2>Methods available to &#064;Actions</h2>
111     * <p>The following are callable from &#064;Actions:
112     * <ul>
113     * <li>Methods on the class passed into {@link Bindings#bind(Object)} if the class is marked as
114     * {@link Bindable}.</li>
115     * <li>Any methods in fields that are marked {@link Bindable} (as in the example).</li>
116     * <li>Any methods in fields that implement {@link BindableModel}.</li>
117     * </ul>
118     * <h2>Disambiguation</h2>
119     * <p>
120     * Strings passed into {@link #call()} are disambiguated as follows:
121     * </p>
122     * <ol>
123     * <li>If the string uniquely identifies a model property or bindable method then 
124     * there is no problem.</li>
125     * <li>If there are two or more possibilities then you must specify the field name in the 
126     * string like "model1.state" or "fileController.close".</li>
127     * <li>You can always use the long form in (2) even if the short form would suffice.</li>
128     * </ol>
129     * 
130     * @see Bound
131     */
132    @Retention(RetentionPolicy.RUNTIME)
133    @Target({ElementType.FIELD})
134    public @interface Action {
135            /**
136             * The name of the method to call when the action occurs. Must be accessible in the 
137             * {@link BindingContext}.
138             */
139            String call();
140    
141            /**
142             * Inner utility class used to wire {@link Action} bindings.
143             */
144            static class Wiring implements BindingWiring {
145                    private static final Logger logger = Logger.getLogger(Action.class);
146                    
147                    /**
148                     * Wires all {@link Action} bindings in the passed context.  
149                     * Called by {@link Bindings#createBindings(BindingContext)} as part of runtime wiring 
150                     * process.
151                     * 
152                     * @param context 
153                     */
154                    public Collection<Binding> wire(BindingContext context) {
155                            List<Field> actions = context.getAnnotatedFields(Action.class);
156                            for (Field field : actions) {
157                                    Action action = field.getAnnotation(Action.class);
158                                    String call = action.call();
159                                    try {
160                                            wire(call, field, context);
161                                    } catch (Exception e) {
162                                            throw new BindingException("could not wire up @Action on " + 
163                                                            field.getName(), e); 
164                                    }               
165                            }
166                            return ImmutableList.of();
167                    }
168                    
169                    /**
170                     * Wires up to any object with an addActionListener method.  Automatically called
171                     * by {@link #wire(BindingContext)}.
172                     * 
173                     * @param call name of an {@link ObjectFieldMethod} in the passed {@link BindingContext}.
174                     * @param field field to bind the call to.
175                     * @param context the {@link BindingContext}
176                     */
177                    private static void wire(String call, Field field, BindingContext context) 
178                                    throws SecurityException, NoSuchMethodException, IllegalArgumentException, 
179                                                    IllegalAccessException, InvocationTargetException {
180                Method aalMethod = field.getType().getMethod("addActionListener", ActionListener.class);
181                            Object actionObject = context.getFieldObject(field, Object.class);
182                            final ObjectFieldMethod ofm = context.getBindableMethod(call);
183                            if (ofm == null) {
184                                    throw new BindingException("could not find bindable method: " + call); 
185                            }
186                            ActionListener actionListener = new ActionListener() {
187                                    public void actionPerformed(ActionEvent e) {
188                                            try {
189                                                    boolean accessible = ofm.getMethod().isAccessible();
190                                                    ofm.getMethod().setAccessible(true);
191                                                    ofm.getMethod().invoke(ofm.getObject());
192                                                    ofm.getMethod().setAccessible(accessible);
193                                            } catch (InvocationTargetException itex) {
194                                                    logger.error("exception during action firing", itex.getCause()); 
195                                            } catch (Exception ex) {
196                                                    logger.error("exception during action firing", ex); 
197                                            }
198                                    }
199                            };
200                            aalMethod.invoke(actionObject, actionListener);
201                    }
202            }
203    }
