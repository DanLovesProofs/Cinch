014    package com.palantir.ptoss.util;
015    
016    import java.io.InterruptedIOException;
017    
018    /**
019     * A class of utility methods to make exception handling cleaner.
020     */
021    public final class Throwables {
022            
023            private Throwables() { /* Static utility methods only. */ }
024            
025            /**
026             * If Throwable is a RuntimeException or Error, rethrow it. If not, throw a 
027             * new PalantirRuntimeException(ex)
028             */
029            public static RuntimeException throwUncheckedException(Throwable ex) {
030                    throwIfInstance(ex, RuntimeException.class);
031                    throwIfInstance(ex, Error.class);
032                    throw createRuntimeException(ex);
033            }       
034            
035            private static RuntimeException createRuntimeException(Throwable ex) {
036                    return createRuntimeException(ex.getMessage(), ex);
037            }
038            
039            private static RuntimeException createRuntimeException(String newMessage, Throwable ex) {
040                    if (ex instanceof InterruptedException || ex instanceof InterruptedIOException) {
041                            Thread.currentThread().interrupt();
042                    }
043                    return new RuntimeException(newMessage, ex);
044            }
045            
046            /**
047             * if (t instanceof K) throw (K)t;
048             * <p>
049             * Note: The runtime type of the thrown exception will be the same as t even if 
050             * clazz is a supertype of t.
051             */
052            @SuppressWarnings("unchecked")
053            public static <K extends Throwable> void throwIfInstance(Throwable t, Class<K> clazz) throws K {
054                    if ((t != null) && clazz.isAssignableFrom(t.getClass())) {              
055                            K kt = (K) t;
056                            throw kt;
057                    }
058            }
059    }
