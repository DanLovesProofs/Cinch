071            /**
072             * Starting at the bottom of a class hierarchy, visit all classes (ancestors) in the hierarchy. Does
073             * not visit interfaces.
074             * @param klass Class to use as the bottom of the class hierarchy
075             * @param visitor Visitor object
076             */
077            public static void visitClassHierarchy(Class<?> klass, Visitor<Class<?>> visitor) {
078                    while (klass != null) {
079                            visitor.visit(klass);
080                            klass = klass.getSuperclass();
081                    }
082            }

084            /**
085             * Given an {@link Object} and a {@link Field} of a known {@link Class} type, get the field.
086             * This will return the value of the field regardless of visibility modifiers (i.e., it will
087             * return the value of private fields.)
088             */
089            public static <T> T getFieldObject(Object object, Field field, Class<T> klass) {
090            try {
091                    boolean accessible = field.isAccessible();
092                    field.setAccessible(true);
093                Object fieldObject = field.get(object);
094                field.setAccessible(accessible);
095                return klass.cast(fieldObject);
096            } catch (IllegalAccessException e) {
097                    // shouldn't happen since we set accessibility above.
098                    return null;
099            }
100        }

102            /**
103             * Returns whether or not the given {@link Field} is final.
104             */
105            public static boolean isFieldFinal(Field field) {
106            int modifiers = field.getModifiers();
107            return Modifier.isFinal(modifiers);
108        }

111             * Returns whether or not the given {@link Field} is static.
112             */
113            public static boolean isFieldStatic(Field field) {
114                    int modifiers = field.getModifiers();
115                    return Modifier.isStatic(modifiers);
116            }

119             * Returns whether or not the given {@link Method} is public.

121            public static boolean isMethodPublic(Method method) {
122            int modifiers = method.getModifiers();
123            return Modifier.isPublic(modifiers);

126            /**
127             * Find a {@link Field} based on the field name.  Will return private fields but will not
128             * look in superclasses.
129             * 
130             * @return null if there is no field found
131             */
132            // TODO (dcervelli) fix for superclasses
133            public static Field getFieldByName(Class<?> klass, String fieldName) {
134            for (Field f : klass.getDeclaredFields()) {
135                    if (f.getName().equals(fieldName)) {
136                            return f;
137                    }
138            }
139            return null;
140        }

142            /**
143             * Gets all inner classes from a given class that are assignable from the target class.
144             * @param klass type to query for inner-classes.
145             * @param targetClass interface or class that inner classes must be assignable from to be
146             * returned.
147             * @return all inner classes in <code>klass</code> that are assignable from 
148             * <code>targetClass</code>
149             * @see Class#isAssignableFrom(Class)
150             * @see Class#getDeclaredClasses()
151             */
152            public static List<Class<?>> getTypesOfType(Class<?> klass, Class<?> targetClass) {
153            List<Class<?>> classes = Lists.newArrayList();
154            for (Class<?> cl : klass.getDeclaredClasses()) {
155                    if (targetClass.isAssignableFrom(cl)) {
156                            classes.add(cl);
157                    }
158            }
159            return classes;
160        }
161    
162            /**
163             * Gets all inner classes assignable from <code>targetClass</code> in the passed class's type
164             * hierarchy.
165             * 
166             * @param klass starting point in the type stack to query for inner classes.
167             * @param targetClass looks for inner classes that are assignable from this type.
168             * @return all inner classes in <code>klass</code>'s type hierarchy assignable from 
169             * <code>targetclass</code>
170             * @see Class#isAssignableFrom(Class)
171             * @see Class#getDeclaredClasses()
172             * @see #getTypesOfType(Class, Class)
173             */
174            public static List<Class<?>> getTypesOfTypeForClassHierarchy(Class<?> klass, final Class<?> targetClass) {
175                    final List<Class<?>> classes = Lists.newArrayList();
176                    visitClassHierarchy(klass, new Visitor<Class<?>>() {
177                            public void visit(Class<?> c) {
178                                    classes.addAll(getTypesOfType(c, targetClass));
179                            }
180                    });
181                    return classes;
182            }

184            /**
185             * Gets all fields from a given class that are assignable from the target class.
186             * @param klass type to query for fields.
187             * @param targetClass interface or class that fields must be assignable from to be
188             * returned.
189             * @return all fields in <code>klass</code> that are assignable from 
190             * <code>targetClass</code>
191             * @see Class#isAssignableFrom(Class)
192             * @see Class#getDeclaredFields()
193             */
194            public static List<Field> getFieldsOfType(Class<?> klass, Class<?> targetClass) {
195            List<Field> fields = Lists.newArrayList();
196            for (Field f : klass.getDeclaredFields()) {
197                    if (targetClass.isAssignableFrom(f.getType())) {
198                            fields.add(f);
199                    }
200            }
201            return fields;
202        }

205             * Gets all fields assignable from <code>targetClass</code> in the passed class's type
206             * hierarchy.
207             * 
208             * @param klass starting point in the type stack to query for fields of the specified type.
209             * @param targetClass looks for fields that are assignable from this type.
210             * @return all fields declared by classes in <code>klass</code>'s type hierarchy assignable from 
211             * <code>targetclass</code>
212             * @see Class#isAssignableFrom(Class)
213             * @see Class#getDeclaredClasses()
214             * @see #getTypesOfType(Class, Class)

216            public static List<Field> getFieldsOfTypeForClassHierarchy(Class<?> klass, final Class<?> targetClass) {

218                    visitClassHierarchy(klass, new Visitor<Class<?>>() {
219                            public void visit(Class<?> c) {

226            /**
227             * Looks up an {@link Enum} value by its {@link String} name.
228             * @param enumType {@link Enum} class to query.
229             * @param value {@link String} name for the {@link Enum} value.
230             * @return the actual {@link Enum} value specified by the passed name.
231             * @see Enum#valueOf(Class, String)
232             */
233            public static Object evalEnum(Class<?> enumType, String value) {
234            try {
235                Method method = enumType.getMethod("valueOf", String.class);
236                method.setAccessible(true);
237                return method.invoke(null, value);
238            } catch (Exception ew) {
239                throw new IllegalArgumentException("could not find enum value: " + value); 
240            }
241        }

243            /**
244             * Checks whether or not the specified {@link Annotation} exists in the passed {@link Object}'s 
245             * class hierarchy.
246             * @param object object to check
247             * @param annotation annotation to look for
248             * @return true is a class in this passed object's type hierarchy is annotated with the 
249             * passed {@link Annotation}
250             */
251            public static boolean isClassAnnotatedForClassHierarchy(Object object, final Class<? extends Annotation> annotation) {
252                    final MutableBoolean bool = new MutableBoolean(false);
253                    visitClassHierarchy(object.getClass(), new Visitor<Class<?>>() {
254                            public void visit(Class<?> klass) {
255                        if (klass.isAnnotationPresent(annotation)) {
256                            bool.setValue(true);
257                        }
258                }
259                    });
260                    return bool.booleanValue();
261            }

263            /**
264             * Returns the list of fields on this class annotated with the passed {@link Annotation}
265             * @param klass checks the {@link Field}s on this class
266             * @param annotation looks for this {@link Annotation}
267             * @return list of all {@link Field}s that are annotated with the specified {@link Annotation}
268             */
269            public static List<Field> getAnnotatedFields(Class<?> klass, Class<? extends Annotation> annotation) {
270            List<Field> annotatedFields = Lists.newArrayList();
271            for (Field f : klass.getDeclaredFields()) {
272                    if (f.isAnnotationPresent(annotation)) {
273                            annotatedFields.add(f);
274                    }
275            }
276            return annotatedFields;
277        }

279            /**
280             * Returns the list of fields on this class or any of its ancestors annotated with the 
281             * passed {@link Annotation}.
282             * @param klass checks the {@link Field}s on this class and its ancestors
283             * @param annotation looks for this {@link Annotation}
284             * @return list of all {@link Field}s that are annotated with the specified {@link Annotation}
285             */
286            public static List<Field> getAnnotatedFieldsForClassHierarchy(Class<?> klass, 
287                            final Class<? extends Annotation> annotation) {
288                    final List<Field> annotatedFields = Lists.newArrayList();
289                    visitClassHierarchy(klass, new Visitor<Class<?>>() {
290                            public void visit(Class<?> c) {
291                                    annotatedFields.addAll(getAnnotatedFields(c, annotation));
292                            }
293                    });
294                    return annotatedFields;
295            }

297            private static List<ObjectFieldMethod> getParameterlessMethods(Object tupleObject, Class<?> klass) {
298            List<ObjectFieldMethod> methods = Lists.newArrayList();
299            for (Method method : klass.getDeclaredMethods()) {
300                    if (method.getParameterTypes().length == 0) {
301                            methods.add(new ObjectFieldMethod(tupleObject, null, method));
302                    }
303            }
304            return methods;
305        }

307            /**
308             * Returns all methods in the passed object's class hierarchy that do no not take parameters
309             * @param object object to query for parameterless methods
310             * @return a list {@link ObjectFieldMethod} tuples mapping the parameterless methods to 
311             * the passed object.
312             */
313            public static List<ObjectFieldMethod> getParameterlessMethodsForClassHierarchy(final Object object) {
314                    final List<ObjectFieldMethod> methods = Lists.newArrayList();
315                    visitClassHierarchy(object.getClass(), new Visitor<Class<?>>() {
316                            public void visit(Class<?> c) {
317                                    methods.addAll(getParameterlessMethods(object, c));
318                            }
319                    });
320                    return methods;
321            }

323            /**
324             * Returns a {@link Function} that will read values from the named field from a passed object.
325             * @param klass type to read values from
326             * @param returnType return type of read field
327             * @param getter name of the field
328             * @return a {@link Function} object that, when applied to an instance of <code>klass</code>, returns the 
329             * of type <code>returnType</code> that resides in field <code>getter</code>
330             */
331            public static <F, T> Function<F, T> getterFunction(final Class<F> klass, final Class<T> returnType, String getter) {
332            try {
333                    BeanInfo beanInfo = Introspector.getBeanInfo(klass);
334                    PropertyDescriptor[] props = beanInfo.getPropertyDescriptors();
335                        Method method = null;
336                        for (PropertyDescriptor descriptor : props) {
337                            if (descriptor.getName().equals(getter)) {
338                                    method = descriptor.getReadMethod();
339                                    break;
340                            }
341                        }
342                        if (method == null) {
343                            throw new IllegalStateException();
344                        }
345                        final Method readMethod = method;
346                        return new Function<F, T>() {
347                            public T apply(F from) {
348                                    try {
349                                            return returnType.cast(readMethod.invoke(from));
350                        } catch (Exception e) {
351                                Throwables.throwUncheckedException(e);
352                                return null;
353                        }
354                            }
355                        };
356            } catch (IntrospectionException e) {
357                    Throwables.throwUncheckedException(e);
358                    return null;
359            }
360            }
